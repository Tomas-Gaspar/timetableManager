38Boolean ADrawOpen(unsigned int w,
39 unsigned int h,
40 const char *file,
41 const char *fontName,
42 ADrawOutputType type,
43 struct ADrawTag *outContext)
44{
45 assert(outContext);
46
47 switch(type)
48 {
49 case ADRAW_FMT_NULL:
50 return NullInit(outContext);
51
52 case ADRAW_FMT_PNG:
53#if !defined(REMOVE_PNG_OUTPUT)
54 return GdoInit(w, h, file, fontName, outContext);
55#else
56 fprintf(stderr, "Built with REMOVE_PNG_OUTPUT; PNG output is not supported\n");
57 return FALSE;
58#endif
59 case ADRAW_FMT_EPS:
60 return PsInit(w, h, file, outContext);
61
62 case ADRAW_FMT_SVG:
63 return SvgInit(w, h, file, outContext);
64
65 default:
66 return FALSE;
67 }
68}

71ADrawColour ADrawGetColour(const char *colour)
72{
73 assert(colour != NULL);
74
75 /* Check if an RGB value has been specified */
76 if(*colour == '#')
77 {
78 unsigned int c = ADRAW_COL_BLACK;
79
80 if(sscanf(&colour[1], "%x", &c) == 1)
81 {
82 return (ADrawColour)c;
83 }
84 }
85 else /* Check for name matches */
86 {
87 static const struct
88 {
89 char *name;
90 ADrawColour col;
91 }
92 colourMap[] =
93 {
94 { "WHITE", ADRAW_COL_WHITE },
95 { "BLACK", ADRAW_COL_BLACK },
96 { "RED", ADRAW_COL_RED },
97 { "ORANGE", ADRAW_COL_ORANGE },
98 { "YELLOW", ADRAW_COL_YELLOW },
99 { "GREEN", ADRAW_COL_GREEN },
100 { "BLUE", ADRAW_COL_BLUE },
101 { "INDIGO", ADRAW_COL_INDIGO },
102 { "VIOLET", ADRAW_COL_VIOLET },
103 { "SILVER", ADRAW_COL_SILVER },
104 { "LIME", ADRAW_COL_LIME },
105 { "GRAY", ADRAW_COL_GRAY },
106 { "OLIVE", ADRAW_COL_OLIVE },
107 { "MAROON", ADRAW_COL_MAROON },
108 { "NAVY", ADRAW_COL_NAVY },
109 { "PURPLE", ADRAW_COL_PURPLE },
110 { "TEAL", ADRAW_COL_TEAL },
111 { "FUCHSIA", ADRAW_COL_FUCHSIA },
112 { "AQUA", ADRAW_COL_AQUA }
113 };
114
115 unsigned int t;
116
117 for(t = 0; t < sizeof(colourMap) / sizeof(colourMap[0]); t++)
118 {
119 if(strcasecmp(colour, colourMap[t].name) == 0)
120 {
121 return colourMap[t].col;
122 }
123 }
124 }
125
126 /* Default to black if all else failed */
127 return ADRAW_COL_BLACK;
128}

131void ADrawComputeArcPoint(float cx, float cy, float w, float h, float degrees,
132 unsigned int *x, unsigned int *y)
133{
134 float rad = (float)((degrees * M_PI) / 180.0f);
135
136 *x = (unsigned int)round(cx + ((w / 2.0f) * cos(rad)));
137 *y = (unsigned int)round(cy + ((h / 2.0f) * sin(rad)));
138}
