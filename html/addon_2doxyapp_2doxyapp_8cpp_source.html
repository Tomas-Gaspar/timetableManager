66 // here we are presented with the symbols found by the code parser
67 void linkableSymbol(int l, const char *sym,Definition *symDef,Definition *context)
68 {
69 QCString ctx;
70 if (context) // the context of the symbol is known
71 {
72 if (context->definitionType()==Definition::TypeMember) // it is inside a member
73 {
74 Definition *parentContext = context->getOuterScope();
75 if (parentContext && parentContext->definitionType()==Definition::TypeClass)
76 // it is inside a member of a class
77 {
78 ctx.sprintf("inside %s %s of %s %s",
79 (dynamic_cast<MemberDef*>(context))->memberTypeName().data(),
80 context->name().data(),
81 (dynamic_cast<ClassDef*>(parentContext))->compoundTypeString().data(),
82 parentContext->name().data());
83 }
84 else if (parentContext==Doxygen::globalScope) // it is inside a global member
85 {
86 ctx.sprintf("inside %s %s",
87 (dynamic_cast<MemberDef*>(context))->memberTypeName().data(),
88 context->name().data());
89 }
90 }
91 if (ctx.isEmpty()) // it is something else (class, or namespace member, ...)
92 {
93 ctx.sprintf("in %s",context->name().data());
94 }
95 }
96 printf("Found symbol %s at line %d of %s %s\n",
97 sym,l,m_fd->getDefFileName().data(),ctx.data());
98 if (symDef && context) // in this case the definition of the symbol is
99 // known to doxygen.
100 {
101 printf("-> defined at line %d of %s\n",
102 symDef->getDefLine(),symDef->getDefFileName().data());
103 }
104 }
105 private:
106 FileDef *m_fd;
107};

109static void findXRefSymbols(FileDef *fd)
110{
111 // get the interface to a parser that matches the file extension
112 auto intf=Doxygen::parserManager->getCodeParser(fd->getDefFileExtension());
113
114 // get the programming language from the file name
115 SrcLangExt lang = getLanguageFromFileName(fd->name());
116
117 // reset the parsers state
118 intf->resetCodeParserState();
119
120 // create a new backend object
121 XRefDummyCodeGenerator *xrefGen = new XRefDummyCodeGenerator(fd);
122
123 // parse the source code
124 intf->parseCode(*xrefGen,
125 0,
126 fileToString(fd->absFilePath()),
127 lang,
128 FALSE,
129 0,
130 fd);
131
132 // dismiss the object.
133 delete xrefGen;
134}

158static void lookupSymbol(const Definition *d)
159{
160 if (d!=Doxygen::globalScope && // skip the global namespace symbol
161 d->name().at(0)!='@' // skip anonymous stuff
162 )
163 {
164 printf("Symbol info\n");
165 printf("-----------\n");
166 printf("Name: %s\n",d->name().data());
167 printf("File: %s\n",d->getDefFileName().data());
168 printf("Line: %d\n",d->getDefLine());
169 // depending on the definition type we can case to the appropriate
170 // derived to get additional information
171 switch (d->definitionType())
172 {
173 case Definition::TypeClass:
174 {
175 const ClassDef *cd = dynamic_cast<const ClassDef*>(d);
176 printf("Kind: %s\n",cd->compoundTypeString().data());
177 }
178 break;
179 case Definition::TypeFile:
180 {
181 const FileDef *fd = dynamic_cast<const FileDef*>(d);
182 printf("Kind: File: #includes %zu other files\n",
183 fd->includeFileList().size());
184 }
185 break;
186 case Definition::TypeNamespace:
187 {
188 const NamespaceDef *nd = dynamic_cast<const NamespaceDef*>(d);
189 printf("Kind: Namespace: contains %zu classes and %zu namespaces\n",
190 nd->getClasses().size(),
191 nd->getNamespaces().size());
192 }
193 break;
194 case Definition::TypeMember:
195 {
196 const MemberDef *md = dynamic_cast<const MemberDef*>(d);
197 printf("Kind: %s\n",md->memberTypeName().data());
198 }
199 break;
200 default:
201 // ignore groups/pages/packages/dirs for now
202 break;
203 }
204 }
205}

207static void lookupSymbols(const QCString &sym)
208{
209 if (!sym.isEmpty())
210 {
211 auto range = Doxygen::symbolMap->find(sym);
212 bool found=false;
213 for (const Definition *def : range)
214 {
215 lookupSymbol(def);
216 found=true;
217 }
218 if (!found)
219 {
220 printf("Unknown symbol\n");
221 }
222 }

225int main(int argc,char **argv)
226{
227 char cmd[256];
228
229 int locArgc = argc;
230
231 if (locArgc == 2)
232 {
233 if (!strcmp(argv[1],"--help"))
234 {
235 printf("Usage: %s [source_file | source_dir]\n",argv[0]);
236 exit(0);
237 }
238 else if (!strcmp(argv[1],"--version"))
239 {
240 printf("%s version: %s\n",argv[0],getFullVersion());
241 exit(0);
242 }
243 }
244
245 if (locArgc!=2)
246 {
247 printf("Usage: %s [source_file | source_dir]\n",argv[0]);
248 exit(1);
249 }
250
251 // initialize data structures
252 initDoxygen();
253
254 // setup the non-default configuration options
255
256 checkConfiguration();
257 adjustConfiguration();
258 // we need a place to put intermediate files
259 Config_updateString(OUTPUT_DIRECTORY,"/tmp/doxygen");
260 // disable html output
261 Config_updateBool(GENERATE_HTML,FALSE);
262 // disable latex output
263 Config_updateBool(GENERATE_LATEX,FALSE);
264 // be quiet
265 Config_updateBool(QUIET,TRUE);
266 // turn off warnings
267 Config_updateBool(WARNINGS,FALSE);
268 Config_updateBool(WARN_IF_UNDOCUMENTED,FALSE);
269 Config_updateBool(WARN_IF_DOC_ERROR,FALSE);
270 // Extract as much as possible
271 Config_updateBool(EXTRACT_ALL,TRUE);
272 Config_updateBool(EXTRACT_STATIC,TRUE);
273 Config_updateBool(EXTRACT_PRIVATE,TRUE);
274 Config_updateBool(EXTRACT_LOCAL_METHODS,TRUE);
275 // Extract source browse information, needed
276 // to make doxygen gather the cross reference info
277 Config_updateBool(SOURCE_BROWSER,TRUE);
278 // In case of a directory take all files on directory and its subdirectories
279 Config_updateBool(RECURSIVE,TRUE);
280
281 // set the input
282 StringVector inputList;
283 inputList.push_back(argv[1]);
284 Config_updateList(INPUT,inputList);
285
286 // parse the files
287 parseInput();
288
289 // iterate over the input files
290 for (const auto &fn : *Doxygen::inputNameLinkedMap)
291 {
292 for (const auto &fd : *fn)
293 {
294 // get the references (linked and unlinked) found in this file
295 findXRefSymbols(fd.get());
296 }
297 }
298
299 // clean up after us
300 Dir().rmdir("/tmp/doxygen");
301
302 while (1)
303 {
304 printf("> Type a symbol name or\n> .list for a list of symbols or\n> .quit to exit\n> ");
305 fgets(cmd,256,stdin);
306 QCString s(cmd);
307 if (s.at(s.length()-1)=='\n') s=s.left(s.length()-1); // strip trailing \n
308 if (s==".list")
309 listSymbols();
310 else if (s==".quit")
311 exit(0);
312 else
313 lookupSymbols(s);
314 }
315}
