150/* Handle a new line of input.
151 * This counts the line number and duplicates the string in case we need
152 * it for error reporting. The line is then returned back for parsing
153 * without the newline characters prefixed.
154 */
155static void newline(const char *text, unsigned int n)
156{
157 lex_linenum++;
158 if(lex_line != NULL)
159 {
160 free(lex_line);
161 }
162
163 lex_line = strdup(text + n);
164 yyless(n);
165}

168/* Trim a multi-line quoted string.
169 * This allows the parsed input quoted strings to span multiple lines of
170 * input but be condensed to only a single line of output e.g.
171 * a->b [label="line 1
172 * line 1 too"];
173 * Will parse to a string such as"line1\n line1 too". This function
174 * will collapse the \n and whitespace into a single space.
175 */
176static char *trimQstring(char *const s)
177{
178 int i = 0, o = 0, skipmode = 0;
179
180 /* Strip leading " */
181 if(s[i] == '\"')
182 {
183 i++;
184 }
185
186 /* Copy body, compacting whitespace after newline sequences */
187 while(s[i] != '\0')
188 {
189 if(s[i] == '\r' || s[i] == '\n' || s[i] == '\f')
190 {
191 skipmode = 1;
192 }
193 else if(!skipmode || !isspace(s[i]))
194 {
195 if(skipmode)
196 {
197 s[o] = ' ';
198 o++;
199 }
200
201 skipmode = 0;
202 s[o] = s[i];
203 o++;
204 }
205
206 i++;
207 }
208
209 /* Null terminate */
210 s[o] = '\0';
211
212 /* Remove trailing " */
213 if(o >= 1 && s[o - 1] == '\"')
214 s[o-1] = '\0';
215
216 return s;
217}
