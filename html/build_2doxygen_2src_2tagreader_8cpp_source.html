366 m_curMember.lineNr = m_locator->lineNr();
367 if (protStr=="protected")

380 {
381 m_curMember.virt = Pure;

1437 case TagClassInfo::Kind::Class: break;

1440 case TagClassInfo::Kind::Interface: ce->spec = Entry::Interface; break;
1441 case TagClassInfo::Kind::Enum: ce->spec = Entry::Enum; break;

1446 case TagClassInfo::Kind::Singleton: ce->spec = Entry::Singleton; break;
1447 case TagClassInfo::Kind::None: // should never happen, means not properly initialized
1448 assert(tci->kind != TagClassInfo::Kind::None);
1449 break;
1450 }
1451 ce->name = tci->name;
1452 if (tci->kind==TagClassInfo::Kind::Protocol)
1453 {
1454 ce->name+="-p";
1455 }
1456 addDocAnchors(ce,tci->docAnchors);
1457 ce->tagInfoData.tagName = m_tagName;

1461 ce->hasTagInfo = TRUE;
1462 ce->id = tci->clangId;
1463 ce->lang = tci->isObjC ? SrcLangExt_ObjC : SrcLangExt_Unknown;
1464 // transfer base class list
1465 ce->extends = tci->bases;
1466 if (!tci->templateArguments.empty())

1470 {
1471 Argument a;
1472 a.type = "class";
1473 a.name = argName.c_str();
1474 al.push_back(a);
1475 }
1476 ce->tArgLists.push_back(al);
1477 }
1478
1479 buildMemberList(ce,tci->members);
1480 root->moveToSubEntryAndKeep(ce);
1481 }
1482 }

1484 // build file list
1485 for (const auto &comp : m_tagFileCompounds)
1486 {

1489 {

1491 fe->section = guessSection(tfi->name);
1492 fe->name = tfi->name;
1493 addDocAnchors(fe,tfi->docAnchors);
1494 fe->tagInfoData.tagName = m_tagName;
1495 fe->tagInfoData.fileName = tfi->filename;
1496 fe->hasTagInfo = TRUE;
1497
1498 QCString fullName = m_tagName+":"+tfi->path+stripPath(tfi->name);
1499 fe->fileName = fullName;
1500 fe->startLine = tfi->lineNr;
1501 //printf("createFileDef() filename=%s\n",qPrint(tfi->filename));
1502 QCString tagid = m_tagName+":"+tfi->path;
1503 std::unique_ptr<FileDef> fd { createFileDef(tagid,
1504 tfi->name,m_tagName,
1505 tfi->filename) };
1506 FileName *mn;
1507 if ((mn=Doxygen::inputNameLinkedMap->find(tfi->name)))
1508 {
1509 mn->push_back(std::move(fd));
1510 }
1511 else
1512 {
1513 mn = Doxygen::inputNameLinkedMap->add(tfi->name,fullName);
1514 mn->push_back(std::move(fd));
1515 }
1516 buildMemberList(fe,tfi->members);
1517 root->moveToSubEntryAndKeep(fe);
1518 }
