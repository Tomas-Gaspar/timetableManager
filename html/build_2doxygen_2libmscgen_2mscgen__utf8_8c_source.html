51static unsigned int clo(char c)
52{
53 unsigned int t = 0;
54
55 while((c & (0x80 >> t)) != 0 && t < 8)
56 {
57 t++;
58 }
59
60 return t;
61}

67Boolean Utf8Decode(const char *s, unsigned int *r, unsigned int *bytes)
68{
69 if((*s & 0x80) == 0)
70 {
71 return FALSE;
72 }
73 else
74 {
75 unsigned int t;
76
77 /* Set default return values */
78 *bytes = clo(*s);
79 *r = 0;
80
81 /* Loop through string, checking for null termination */
82 for(t = 0; t < *bytes && s[t] != '\0'; t++)
83 {
84 /* Shift up previous bits */
85 *r <<= 6;
86
87 /* Add the required bits */
88 if(t == 0)
89 {
90 *r |= s[t] & (0xff >> (*bytes + 1));
91 }
92 else
93 {
94 *r |= s[t] & 0x3f;
95 }
96 }
97
98 /* Success if no NULL was encountered */
99 return t == *bytes;
100 }
101}
