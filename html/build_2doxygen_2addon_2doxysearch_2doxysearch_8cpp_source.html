51bool dirExists(const std::string& dirName)

57
58 if (ftyp & FILE_ATTRIBUTE_DIRECTORY)
59 return true; // this is a directory!

61 struct stat sb;

74static std::string uriDecode(const std::string & sSrc)
75{
76 // Note from RFC1630: "Sequences which start with a percent
77 // sign but are not followed by two hexadecimal characters
78 // (0-9, A-F) are reserved for future extension"
79
80 const unsigned char * pSrc = (const unsigned char *)sSrc.c_str();
81 const size_t SRC_LEN = sSrc.length();
82 const unsigned char * const SRC_END = pSrc + SRC_LEN;
83 // last decodable '%'
84 const unsigned char * const SRC_LAST_DEC = SRC_END - 2;
85
86 char * const pStart = new char[SRC_LEN];
87 char * pEnd = pStart;
88
89 while (pSrc < SRC_LAST_DEC)
90 {
91 if (*pSrc == '%') // replace %2A with corresponding ASCII character
92 {
93 char dec1, dec2;
94 unsigned char c1=*(pSrc+1);
95 unsigned char c2=*(pSrc+2);
96 if (-1 != (dec1 = HEX2DEC(c1))
97 && -1 != (dec2 = HEX2DEC(c2)))
98 {
99 *pEnd++ = (dec1 << 4) + dec2;
100 pSrc += 3;
101 continue;
102 }
103 }
104 else if (*pSrc == '+') // replace '+' with space
105 {
106 *pEnd++ = ' '; pSrc++;
107 continue;
108 }
109 *pEnd++ = *pSrc++;
110 }
111
112 // the last 2- chars
113 while (pSrc < SRC_END) *pEnd++ = *pSrc++;
114
115 std::string sResult(pStart, pEnd);
116 delete [] pStart;
117 return sResult;
118}

185static bool insideRange(const std::vector<Range> &ranges,size_t start,size_t len)
186{
187 for (std::vector<Range>::const_iterator it = ranges.begin();
188 it!=ranges.end(); ++it
189 )
190 {
191 Range r = *it;
192 if (start>=r.start && start+len<r.end)
193 {
194 return true;
195 }
196 }
197 return false;

204static void highlighter(const std::string &s,
205 const std::vector<std::string> &words,
206 std::vector<Fragment> &fragments)
207{
208 const std::string spanStart="<span class=\"hl\">";
209 const std::string spanEnd="</span>";
210 const std::string dots="...";
211 const size_t fragLen = 60;
212 size_t sl=s.length();
213
214 // find positions of words in s
215 int j=0;
216 std::vector<WordPosition> positions;
217 for (std::vector<std::string>::const_iterator it=words.begin();
218 it!=words.end();
219 ++it,++j
220 )
221 {
222 size_t pos=0;
223 size_t i;
224 std::string word = *it;
225 while ((i=s.find(word,pos))!=std::string::npos)
226 {
227 positions.push_back(WordPosition(i,j));
228 pos=i+word.length();
229 }
230 }
231 // sort on position
232 std::sort(positions.begin(),positions.end(),WordPosition_less());
233 // get fragments around words
234 std::vector<Range> ranges;
235 for (std::vector<WordPosition>::const_iterator it=positions.begin();
236 it!=positions.end();
237 ++it)
238 {
239 WordPosition wp = *it;
240 std::string w = words[wp.index];
241 size_t i=wp.start;
242 size_t wl=w.length();
243 if (!insideRange(ranges,i,wl))
244 {
245 if (wl>fragLen)
246 {
247 fragments.push_back(Fragment(spanStart+w+spanEnd,1));
248 ranges.push_back(Range(i,i+wl));
249 }
250 else
251 {
252 std::string startFragment,endFragment;
253 int bi=static_cast<int>(i)-static_cast<int>((fragLen-wl)/2);
254 size_t ei=i+wl+(fragLen-wl)/2;
255 int occ=0;
256 if (bi<0) { ei-=bi; bi=0; } else startFragment=dots;
257 if (ei>sl) { ei=sl; } else endFragment=dots;
258 while (bi>0 && !isspace(s[bi])) bi--; // round to start of the word
259 while (ei<sl && !isspace(s[ei])) ei++; // round to end of the word
260 // highlight any word in s between indexes bi and ei
261 std::string fragment=startFragment;
262 size_t pos=bi;
263 for (std::vector<WordPosition>::const_iterator it2=positions.begin();
264 it2!=positions.end();
265 ++it2)
266 {
267 WordPosition wp2 = *it2;
268 std::string w2 = words[wp2.index];
269 size_t wl2 = w2.length();
270 if (wp2.start>=bi && wp2.start+wl2<=ei) // word is inside the range!
271 {
272 fragment+=s.substr(pos,wp2.start-pos)+
273 spanStart+
274 s.substr(wp2.start,wl2)+
275 spanEnd;
276 pos=wp2.start+wl2;
277 occ++;
278 }
279 }
280 fragment+=s.substr(pos,ei-pos)+endFragment;
281 fragments.push_back(Fragment(fragment,occ));
282 ranges.push_back(Range(bi,ei));
283 }
284 }
285 }
286 std::sort(fragments.begin(),fragments.end(),Fragment_greater());
287}

290static std::string escapeString(const std::string &s)
291{
292 std::stringstream dst;
293 for (size_t i=0;i<s.length();i++)
294 {
295 char ch = s[i];
296 switch (ch)
297 {
298 case '\"': dst << "\\\""; break;
299 default: dst << ch; break;
300 }
301 }
302 return dst.str();
303}

319int main(int argc,char **argv)
320{
321 // process inputs that were passed to us via QUERY_STRING
322 std::string callback;
323 try
324 {
325 std::string queryString;
326 if (argc == 1)
327 {
328 const char *queryEnv = getenv("QUERY_STRING");
329 if (queryEnv)
330 {
331 queryString = queryEnv;
332 }
333 else
334 {
335 usage(argv[0]);
336 }
337 }
338 else if (argc == 2)
339 {
340 if (std::string(argv[1])=="-h" || std::string(argv[1])=="--help")
341 {
342 usage(argv[0],0);
343 }
344 else if (std::string(argv[1])=="-v" || std::string(argv[1])=="--version")
345 {
346 std::cerr << argv[0] << " version: " << getFullVersion() << std::endl;
347 exit(0);
348 }
349 else
350 {
351 queryString = argv[1];
352 }
353 }
354 else
355 {
356 usage(argv[0]);
357 }
358
359 std::cout << "Content-Type:application/javascript;charset=utf-8\r\n\n";
360 // parse query string
361 std::vector<std::string> parts = split(queryString,'&');
362 std::string searchFor,callback;
363 int num=1,page=0;
364 for (std::vector<std::string>::const_iterator it=parts.begin();it!=parts.end();++it)
365 {
366 std::vector<std::string> kv = split(*it,'=');
367 if (kv.size()==2)
368 {
369 std::string val = uriDecode(kv[1]);
370 if (kv[0]=="q") searchFor = val;
371 else if (kv[0]=="n") num = fromString<int>(val);
372 else if (kv[0]=="p") page = fromString<int>(val);
373 else if (kv[0]=="cb") callback = val;
374 }
375 }
376
377 std::string indexDir = "doxysearch.db";
378
379 if (queryString=="test") // user test
380 {
381 bool dbOk = dirExists(indexDir);
382 if (dbOk)
383 {
384 std::cout << "Test successful.";
385 }
386 else
387 {
388 std::cout << "Test failed: cannot find search index " << indexDir;
389 }
390 exit(0);
391 }
392
393 // create query
394 Xapian::Database db(indexDir);
395 Xapian::Enquire enquire(db);
396
397 std::vector<std::string> words = split(searchFor,' ');
398 Xapian::QueryParser parser;
399 parser.set_database(db);
400 parser.set_default_op(Xapian::Query::OP_AND);
401 parser.set_stemming_strategy(Xapian::QueryParser::STEM_ALL);
402 Xapian::termcount max_expansion=100;
403#if (XAPIAN_MAJOR_VERSION==1) && (XAPIAN_MINOR_VERSION==2)
404 parser.set_max_wildcard_expansion(max_expansion);
405#else
406 parser.set_max_expansion(max_expansion,Xapian::Query::WILDCARD_LIMIT_MOST_FREQUENT);
407#endif
408 Xapian::Query query=parser.parse_query(searchFor,
409 Xapian::QueryParser::FLAG_DEFAULT |
410 Xapian::QueryParser::FLAG_WILDCARD |
411 Xapian::QueryParser::FLAG_PHRASE |
412 Xapian::QueryParser::FLAG_PARTIAL
413 );
414 enquire.set_query(query);
415
416 // get results
417 Xapian::MSet matches = enquire.get_mset(page*num,num);
418 unsigned int hits = matches.get_matches_estimated();
419 unsigned int offset = page*num;
420 unsigned int pages = num>0 ? (hits+num-1)/num : 0;
421 if (offset>hits) offset=hits;
422 if (offset+num>hits) num=hits-offset;
423
424 // write results as JSONP
425 std::cout << callback.c_str() << "(";
426 std::cout << "{" << std::endl
427 << " \"hits\":" << hits << "," << std::endl
428 << " \"first\":" << offset << "," << std::endl
429 << " \"count\":" << num << "," << std::endl
430 << " \"page\":" << page << "," << std::endl
431 << " \"pages\":" << pages << "," << std::endl
432 << " \"query\": \"" << escapeString(searchFor) << "\"," << std::endl
433 << " \"items\":[" << std::endl;
434 // foreach search result
435 unsigned int o = offset;
436 for (Xapian::MSetIterator i = matches.begin(); i != matches.end(); ++i,++o)
437 {
438 std::vector<Fragment> hl;
439 Xapian::Document doc = i.get_document();
440 highlighter(doc.get_value(FIELD_DOC),words,hl);
441 std::cout << " {\"type\": \"" << doc.get_value(FIELD_TYPE) << "\"," << std::endl
442 << " \"name\": \"" << doc.get_value(FIELD_NAME) << escapeString(doc.get_value(FIELD_ARGS)) << "\"," << std::endl
443 << " \"tag\": \"" << doc.get_value(FIELD_TAG) << "\"," << std::endl
444 << " \"url\": \"" << doc.get_value(FIELD_URL) << "\"," << std::endl;
445 std::cout << " \"fragments\":[" << std::endl;
446 int c=0;
447 bool first=true;
448 for (std::vector<Fragment>::const_iterator it = hl.begin();it!=hl.end() && c<3;++it,++c)
449 {
450 if (!first) std::cout << "," << std::endl;
451 std::cout << " \"" << escapeString((*it).text) << "\"";
452 first=false;
453 }
454 if (!first) std::cout << std::endl;
455 std::cout << " ]" << std::endl;
456 std::cout << " }";
457 if (o<offset+num-1) std::cout << ",";
458 std::cout << std::endl;
459 }
460 std::cout << " ]" << std::endl << "})" << std::endl;
461 }
462 catch (const Xapian::Error &e) // Xapian exception
463 {
464 showError(callback,e.get_description());
465 }
466 catch (...) // Any other exception
467 {
468 showError(callback,"Unknown Exception!");
469 exit(1);
470 }
471 return 0;
472}
