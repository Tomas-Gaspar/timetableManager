131void InputString::updateDefault()
132{
133 {
134 if (m_str==m_default || !m_lab->isEnabled())
135 {
136 m_lab->setText(QString::fromLatin1("<qt>")+m_id+QString::fromLatin1("</qt>"));
137 }
138 else
139 {
140 m_lab->setText(QString::fromLatin1("<qt><font color='red'>")+m_id+QString::fromLatin1("</font></qt>"));
141 }
142 if (m_im)
143 {
144 if (m_str.isEmpty())
145 {
146 m_im->setText(tr("No Project logo selected."));
147 }
148 else
149 {
150 QFile Fout(m_str);
151 if(!Fout.exists())
152 {
153 m_im->setText(tr("Sorry, cannot find file(")+m_str+QString::fromLatin1(");"));
154 }
155 else
156 {
157 QPixmap pm(m_str);
158 if (!pm.isNull())
159 {
160 m_im->setPixmap(pm.scaledToHeight(55,Qt::SmoothTransformation));
161 }
162 else
163 {
164 m_im->setText(tr("Sorry, no preview available (")+m_str+QString::fromLatin1(");"));
165 }
166 }
167 }
168 }
169 if (m_le && m_le->text()!=m_str) m_le->setText( m_str );
170 emit changed();
171 }

186void InputString::browseFile()
187{
188 QString path = QFileInfo(MainWindow::instance().configFileName()).path();
189 QString fileName = QFileDialog::getOpenFileName(&MainWindow::instance(),
190 tr("Select file"),path);
191 if (!fileName.isNull())
192 {
193 QDir dir(path);
194 if (!MainWindow::instance().configFileName().isEmpty() && dir.exists())
195 {
196 fileName = m_absPath ? fileName : dir.relativeFilePath(fileName);
197 }
198 setValue(fileName);
199 }
