21InputBool::InputBool( QGridLayout *layout, int &row,
22 const QString &id, bool checked,
23 const QString &docs )
24 : m_default(checked), m_docs(docs), m_id(id)
25{
26 m_lab = new HelpLabel(id);
27 m_cb = new QCheckBox;
28 layout->addWidget(m_lab,row, 0);
29 layout->addWidget(m_cb,row, 1);
30 m_enabled = true;
31 m_state=!checked; // force update
32 setValue(checked);
33 connect( m_cb, SIGNAL(toggled(bool)), SLOT(setValue(bool)) );
34 connect( m_lab, SIGNAL(enter()), SLOT(help()) );
35 connect( m_lab, SIGNAL(reset()), SLOT(reset()) );
36 row++;
37}

91bool InputBool::convertToBool(const QVariant &value,bool &isValid)
92{
93 QString v = value.toString().toLower();
94 if (v==QString::fromLatin1("yes") || v==QString::fromLatin1("true") ||
95 v==QString::fromLatin1("1") || v==QString::fromLatin1("all"))
96 {
97 isValid = true;
98 return true;
99 }
100 else if (v==QString::fromLatin1("no") || v==QString::fromLatin1("false") ||
101 v==QString::fromLatin1("0") || v==QString::fromLatin1("none"))
102 {
103 isValid = true;
104 return false;
105 }
106 else
107 {
108 isValid = false;
109 return false;
110 }
111}

113void InputBool::update()
114{
115 bool isValid=false;
116 bool b = convertToBool(m_value,isValid);
117 if (isValid)
118 {
119 m_state = b;
120 }
121 else
122 {
123 config_warn("argument '%s' for option %s is not a valid boolean value."
124 " Using the default: %s!",qPrintable(m_value.toString()),qPrintable(m_id),m_default?"YES":"NO");
125 m_state = m_default;
126 }
127 m_cb->setChecked( m_state );
128 updateDefault();
129 updateDependencies();
130}
