69static char asciiToLower(char in) {
70 if (in <= 'Z' && in >= 'A')
71 return in - ('Z' - 'z');
72 return in;
73}

75static int labelToEnumValue(const QCString &l)
76{
77 std::string s = l.str();
78 std::transform(s.begin(),s.end(),s.begin(),asciiToLower);
79 auto it = s_labels.find(s);
80 return (it!=s_labels.end()) ? it->second : 0;
81}

83int Debug::setFlag(const QCString &lab)
84{
85 int retVal = labelToEnumValue(lab);
86 curMask = static_cast<DebugMask>(curMask | retVal);
87 return retVal;
88}

90void Debug::clearFlag(const QCString &lab)
91{
