100static const char *findAttrib(const struct MscAttribTag *attr, MscAttribType a)
101{
102 while(attr != NULL && attr->type != a)
103 {
104 attr = attr->next;
105 }
106
107 if(attr != NULL)
108 {
109 return attr->value;
110 }
111 else
112 {
113 return NULL;
114 }

153{
154 struct MscOptTag *tail = newHead;

232/* MscLinkEntity
233 * Link some entity onto a list, possibly producing a new head element.
234 */
235struct MscEntityListTag *MscLinkEntity(struct MscEntityListTag *list,
236 struct MscEntityTag *elem)
237{
238 /* Check if the list has been allocated or not */
239 if(list == NULL)
240 {
241 list = zalloc_s(sizeof(struct MscEntityListTag));
242 }
243
244 /* Check for an empty list */
245 if(list->head == NULL)
246 {
247 list->head = list->tail = elem;
248 }
249 else
250 {
251 /* Add to tail */
252 list->tail->next = elem;
253 list->tail = elem;
254 }
255
256 /* Increment count of elements */
257 list->elements++;
258
259 return list;
260}

281 */
282struct MscArcTag *MscAllocArc(char *srcEntity,
283 char *dstEntity,
284 MscArcType type,
285 unsigned int inputLine)
286{
287 struct MscArcTag *a = malloc_s(sizeof(struct MscArcTag));
288
289 /* A discontinuity arcs are not between entities */
290 if(type == MSC_ARC_DISCO)
291 {
292 assert(srcEntity == NULL && dstEntity == NULL);
293 }
294
295 a->inputLine = inputLine;
296 a->src = srcEntity;
297 a->dst = dstEntity;
298 a->type = type;
299 a->next = NULL;
300 a->attr = NULL;
301
302 return a;
303}

306/* MscLinkArc
307 * Link some entity onto a list, possibly producing a new head element.
308 */
309struct MscArcListTag *MscLinkArc(struct MscArcListTag *list,
310 struct MscArcTag *elem)
311{
312 /* Check if the list has been allocated or not */
313 if(list == NULL)
314 {
315 list = zalloc_s(sizeof(struct MscArcListTag));
316 }
317
318 /* Check for an empty list */
319 if(list->head == NULL)
320 {
321 list->head = list->tail = elem;
322 }
323 else
324 {
325 /* Add to tail */
326 list->tail->next = elem;
327 list->tail = elem;
328 }
329
330 /* Increment count of elements */
331 list->elements++;
332 if(elem->type == MSC_ARC_PARALLEL)
333 {
334 /* A parallel arc is a place holder, and indicates the next arc
335 * is on the same line. It also needs to account itself, so subtract
336 * two here.
337 */
338 list->parElements += 2;
339 }
340
341 return list;
342}

500void MscFree(struct MscTag *m)
501{
502 struct MscOptTag *opt = m->optList;
503 struct MscEntityTag *entity = m->entityList->head;
504 struct MscArcTag *arc = m->arcList->head;
505
506 while(opt)
507 {
508 struct MscOptTag *next = opt->next;
509
510 free(opt->value);
511 free(opt);
512
513 opt = next;
514 }
515
516 while(entity)
517 {
518 struct MscEntityTag *next = entity->next;
519
520 freeAttribList(entity->attr);
521 free(entity->label);
522 free(entity);
523
524 entity = next;
525 }
526
527 while(arc)
528 {
529 struct MscArcTag *next = arc->next;
530
531 freeAttribList(arc->attr);
532 free(arc->dst);
533 free(arc->src);
534 free(arc);
535
536 arc = next;
537 }
538
539 free(m->entityList);
540 free(m->arcList);
541 free(m);
542}

652{
653 struct MscEntityTag *entity;

772Boolean MscGetOptAsBoolean(struct MscTag *m, MscOptType type, Boolean *const b)
773{
774 struct MscOptTag *opt = MscFindOpt(m->optList, type);
775
776 if(opt != NULL)
777 {
778 const char *v = opt->value;
779
780 if(strcasecmp(v, "true") == 0 || strcasecmp(v, "yes") == 0 ||
781 strcasecmp(v, "on") == 0 || strcasecmp(v, "1") == 0)
782 {
783 *b = TRUE;
784 return TRUE;
785 }
786 else if(strcasecmp(v, "false") == 0 || strcasecmp(v, "no") == 0 ||
787 strcasecmp(v, "off") == 0 || strcasecmp(v, "0") == 0)
788 {
789 *b = FALSE;
790 return TRUE;
791 }
792 else
793 {
794 fprintf(stderr, "Warning: Unrecognised boolean option value '%s'. Valid values are 'true',\n"
795 " 'false', 'yes', 'no', 'on', 'off', '1' and '0'.\n",
796 v);
797 return FALSE;
798 }
799 }
800
801 return FALSE;
802}
