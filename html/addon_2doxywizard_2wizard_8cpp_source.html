228void hsl2rgb(double h,double s,double l,
229 double *pRed,double *pGreen,double *pBlue)
230{
231 double v;
232 double r,g,b;
233
234 r = l; // default to gray
235 g = l;
236 b = l;
237 v = (l <= 0.5) ? (l * (1.0 + s)) : (l + s - l * s);
238 if (v > 0)
239 {
240 double m;
241 double sv;
242 int sextant;
243 double fract, vsf, mid1, mid2;
244
245 m = l + l - v;
246 sv = (v - m ) / v;
247 h *= 6.0;
248 sextant = (int)h;
249 fract = h - sextant;
250 vsf = v * sv * fract;
251 mid1 = m + vsf;
252 mid2 = v - vsf;
253 switch (sextant)
254 {
255 case 0:
256 r = v;
257 g = mid1;
258 b = m;
259 break;
260 case 1:
261 r = mid2;
262 g = v;
263 b = m;
264 break;
265 case 2:
266 r = m;
267 g = v;
268 b = mid1;
269 break;
270 case 3:
271 r = m;
272 g = mid2;
273 b = v;
274 break;
275 case 4:
276 r = mid1;
277 g = m;
278 b = v;
279 break;
280 case 5:
281 r = v;
282 g = m;
283 b = mid2;
284 break;
285 }
286 }
287 *pRed = r;
288 *pGreen = g;
289 *pBlue = b;
290}

294void TuneColorDialog::updateImage(int hue,int sat,int gam)
295{
296 QImage coloredImg(m_image->width(),m_image->height(),QImage::Format_RGB32);
297 uint *srcPixel = (uint *)m_image->scanLine(0);
298 uint *dstPixel = (uint *)coloredImg.scanLine(0);
299 uint nrPixels = coloredImg.width()*coloredImg.height();
300 for (uint i=0;i<nrPixels;i++,srcPixel++,dstPixel++)
301 {
302 QColor c = QColor::fromRgb(*srcPixel);
303 double r,g,b;
304 hsl2rgb(hue/359.0, sat/255.0, pow(c.green()/255.0,gam/100.0),&r,&g,&b);
305 *dstPixel = qRgb((int)(r*255.0),(int)(g*255.0),(int)(b*255.0));
306 }
307 m_imageLab->setPixmap(QPixmap::fromImage(coloredImg));
308 m_hue = hue;
309 m_sat = sat;
310 m_gam = gam;
311}

344void ColorPicker::paintEvent(QPaintEvent*)
345{
346 int w = width() - 5;
347
348 QRect r(0, foff, w, height() - 2*foff);
349 int wi = r.width() - 2;
350 int hi = r.height() - 2;
351 if (!m_pix || m_pix->height() != hi || m_pix->width() != wi)
352 {
353 delete m_pix;
354 QImage img(wi, hi, QImage::Format_RGB32);
355 int y;
356 uint *pixel = (uint *) img.scanLine(0);
357 for (y = 0; y < hi; y++)
358 {
359 const uint *end = pixel + wi;
360 int yh = y2hue(y+coff);
361 int ys = y2sat(y+coff);
362 int yg = y2gam(y+coff);
363 while (pixel < end)
364 {
365 QColor c;
366 c.setHsv(yh, ys, (int)(255*pow(0.7,yg/100.0)));
367 *pixel = c.rgb();
368 ++pixel;
369 }
370 }
371 m_pix = new QPixmap(QPixmap::fromImage(img));
372 }
373 QPainter p(this);
374 p.drawPixmap(1, coff, *m_pix);
375 const QPalette &g = palette();
376 qDrawShadePanel(&p, r, g, true);
377 p.setPen(g.windowText().color());
378 p.setBrush(g.windowText());
379 QPolygon a;
380 int y = m_mode==Hue ? hue2y(m_hue) :
381 m_mode==Saturation ? sat2y(m_sat) :
382 gam2y(m_gam);
383 a.setPoints(3, w, y, w+5, y+5, w+5, y-5);
384 p.eraseRect(w, 0, 5, height());
385 p.drawPolygon(a);
386}

473 int d = height() - 2*coff - 1;
