31InputInt::InputInt( QGridLayout *layout,int &row,
32 const QString & id,

61void InputInt::setValue(int val)
62{
63 int newVal = val;
64 newVal = qMax(m_minVal,newVal);
65 newVal = qMin(m_maxVal,newVal);
66 if (val != newVal)
67 {
68 config_warn("argument '%d' for option %s is not a valid number in the range [%d..%d]!"
69 " Using the default: %d!\n",val,qPrintable(m_id),m_minVal,m_maxVal,m_default);
70 newVal = m_default;
71 }
72 if (newVal!=m_val)
73 {
74 m_val = newVal;
75 m_sp->setValue(newVal);
76 m_value = m_val;
77 updateDefault();
78 }
79}
