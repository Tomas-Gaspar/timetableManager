422void MainWindow::loadSettings()
423{
424 QVariant geometry = m_settings.value(QString::fromLatin1("main/geometry"));
425 QVariant state = m_settings.value(QString::fromLatin1("main/state"));
426 QVariant wizState = m_settings.value(QString::fromLatin1("wizard/state"));
427 QVariant loadSettings = m_settings.value(QString::fromLatin1("wizard/loadsettings"));
428 QVariant workingDir = m_settings.value(QString::fromLatin1("wizard/workingdir"));
429
430 if (!geometry.isNull()) restoreGeometry(geometry.toByteArray());
431 if (!state.isNull()) restoreState (state.toByteArray());
432 if (!wizState.isNull()) m_wizard->restoreState(wizState.toByteArray());
433 if (!loadSettings.isNull() && loadSettings.toBool())
434 {
435 m_expert->loadSettings(&m_settings);
436 if (!workingDir.isNull() && QDir(workingDir.toString()).exists())
437 {
438 setWorkingDir(workingDir.toString());
439 }
440 }
441
442 /* due to prepend use list in reversed order */
443 for (int i=MAX_RECENT_FILES;i>=0;i--)
444 {
445 QString entry = m_settings.value(QString::fromLatin1("recent/config%1").arg(i)).toString();
446 if (!entry.isEmpty() && QFileInfo(entry).exists())
447 {
448 addRecentFileList(entry);
449 }
450 }
451 updateRecentFile();
452
453}

481void MainWindow::addRecentFileList(const QString &fileName)
482{
483 int i=m_recentFiles.indexOf(fileName);
484 if (i!=-1) m_recentFiles.removeAt(i);
485
486 // not found
487 if (m_recentFiles.count() < MAX_RECENT_FILES) // append
488 {
489 m_recentFiles.prepend(fileName);
490 }
491 else // add + drop last item
492 {
493 m_recentFiles.removeLast();
494 m_recentFiles.prepend(fileName);
495 }
496 m_clearRecent->setEnabled(m_recentFiles.count()>0);
497 m_recentMenu->setEnabled(m_recentFiles.count()>0);
498 m_settings.sync();
499}
500void MainWindow::updateRecentFile(void)
501{
502 m_recentMenu->clear();
503 int i=0;
504 foreach( QString str, m_recentFiles )
505 {
506 m_recentMenu->addAction(str);
507 m_settings.setValue(QString::fromLatin1("recent/config%1").arg(i++),str);
508 }
509 for (;i<MAX_RECENT_FILES;i++)
510 {
511 m_settings.setValue(QString::fromLatin1("recent/config%1").arg(i),QString::fromLatin1(""));
512 }
513 m_clearRecent->setEnabled(m_recentFiles.count()>0);
514 m_recentMenu->setEnabled(m_recentFiles.count()>0);
515 m_settings.sync();
516}

526void MainWindow::runDoxygen()
527{
528 if (!m_running)
529 {
530 QString doxygenPath;
531#if defined(Q_OS_MACX)
532 doxygenPath = qApp->applicationDirPath()+QString::fromLatin1("/../Resources/");
533 qDebug() << tr("Doxygen path: ") << doxygenPath;
534 if ( !QFile(doxygenPath + QString::fromLatin1("doxygen")).exists() )
535 {
536 // No Doxygen binary in the resources, if there is a system Doxygen binary, use that instead
537 if ( QFile(QString::fromLatin1("/usr/local/bin/doxygen")).exists() )
538 {
539 doxygenPath = QString::fromLatin1("/usr/local/bin/");
540 }
541 else
542 {
543 qDebug() << tr("Can't find the doxygen command, make sure it's in your $$PATH");
544 doxygenPath = QString::fromLatin1("");
545 }
546 }
547 qDebug() << tr("Getting doxygen from: ") << doxygenPath;
548#endif
549
550 m_runProcess->setReadChannel(QProcess::StandardOutput);
551 m_runProcess->setProcessChannelMode(QProcess::MergedChannels);
552 m_runProcess->setWorkingDirectory(m_workingDir->text());
553 QStringList env=QProcess::systemEnvironment();
554 // set PWD environment variable to m_workingDir
555 env.replaceInStrings(QRegularExpression(QString::fromLatin1("^PWD=(.*)"),QRegularExpression::CaseInsensitiveOption),
556 QString::fromLatin1("PWD=")+m_workingDir->text());
557 m_runProcess->setEnvironment(env);
558
559 QStringList args;
560#if QT_VERSION < QT_VERSION_CHECK(5, 15, 0)
561 QStringList runOptions = m_runOptions->text().split(QLatin1Char(' '),QString::SkipEmptyParts);
562#else
563 QStringList runOptions = m_runOptions->text().split(QLatin1Char(' '),Qt::SkipEmptyParts);
564#endif
565
566 args << runOptions;
567 args << QString::fromLatin1("-b"); // make stdout unbuffered
568 args << QString::fromLatin1("-"); // read config from stdin
569
570 m_outputLog->clear();
571 text1 = QString::fromLatin1("");
572 m_runProcess->start(doxygenPath + QString::fromLatin1("doxygen"), args);
573
574 if (!m_runProcess->waitForStarted())
575 {
576 m_outputLog->append(APPQT(QString::fromLatin1("*** Failed to run doxygen\n")));
577 return;
578 }
579 QTextStream t(m_runProcess);
580 m_expert->writeConfig(t,false,false);
581 t.flush();
582 m_runProcess->closeWriteChannel();
583
584 if (m_runProcess->state() == QProcess::NotRunning)
585 {
586 m_outputLog->append(APPQT(QString::fromLatin1("*** Failed to run doxygen\n")));
587 }
588 else
589 {
590 m_saveLog->setEnabled(false);
591 m_running=true;
592 m_run->setText(tr("Stop doxygen"));
593 m_runStatus->setText(tr("Status: running"));
594 m_timer->start(1000);
595 }
596 }
597 else
598 {
599 m_running=false;
600 m_run->setText(tr("Run doxygen"));
601 m_runStatus->setText(tr("Status: not running"));
602 m_runProcess->kill();
603 m_timer->stop();
604 //updateRunnable(m_workingDir->text());
605 }
606}

726bool MainWindow::discardUnsavedChanges(bool saveOption)
727{
728 if (m_modified)
729 {
730 QMessageBox::StandardButton button;
731 if (saveOption)
