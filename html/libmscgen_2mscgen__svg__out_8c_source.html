151static void arcPoint(float cx,
152 float cy,
153 float w,
154 float h,
155 float a,
156 unsigned int *x,
157 unsigned int *y)
158{
159 float rad = (float)((a * M_PI) / 180.0f);
160
161 /* Compute point, noting this is for SVG co-ordinate system */
162 *x = (unsigned int)round(cx + ((w / 2.0f) * cos(rad)));
163 *y = (unsigned int)round(cy + ((h / 2.0f) * sin(rad)));
164}

169static void writeEscaped(struct ADrawTag *ctx, const char *string)
170{
171 FILE *f = getSvgFile(ctx);
172
173 while(*string != '\0')
174 {
175 unsigned int code, bytes;
176
177 switch(*string)
178 {
179 case '<': fprintf(f, "&lt;"); break;
180 case '>': fprintf(f, "&gt;"); break;
181 case '"': fprintf(f, "&quot;"); break;
182 case '&': fprintf(f, "&amp;"); break;
183 default:
184 if(Utf8Decode(string, &code, &bytes))
185 {
186 fprintf(f, "&#x%x;", code);
187 string += bytes - 1;
188 }
189 else
190 {
191 fputc(*string, f);
192 }
193 break;
194
195 }
196
197 string++;
198 }
199}

248unsigned int SvgTextWidth(struct ADrawTag *ctx,
249 const char *string)
250{
251 unsigned long width = 0;
252
253 while(*string != '\0')
254 {
255 int i = *string & 0xff;
256 unsigned long w = SvgHelvetica.widths[i];
257
258 /* Ignore undefined characters */
259 width += w > 0 ? w : 0;
260
261 string++;
262 }
263
264 return getSpace(ctx, width);
265}

299void SvgTextR(struct ADrawTag *ctx,
300 unsigned int x,
301 unsigned int y,
302 const char *string)
303{
304 SvgContext *context = getSvgCtx(ctx);
305
306 svgRect(ctx, getSvgBgPen(ctx), x - 2, y - SvgTextHeight(ctx) + 1, x + SvgTextWidth(ctx, string), y - 1);
307
308 y += getSpace(ctx, SvgHelvetica.descender);
309
310 fprintf(getSvgFile(ctx),
311 "<text x=\"%u\" y=\"%u\" textLength=\"%u\" font-family=\"Helvetica\" font-size=\"%u\" fill=\"%s\">\n",
312 x - 1, y, SvgTextWidth(ctx, string), context->fontPoints, context->penColName);
313 writeEscaped(ctx, string);
314 fprintf(getSvgFile(ctx), "\n</text>\n");
315}

518Boolean SvgInit(unsigned int w,
519 unsigned int h,
520 const char *file,
521 struct ADrawTag *outContext)
522{
523 SvgContext *context;
524
525 /* Create context */
526 context = outContext->internal = malloc_s(sizeof(SvgContext));
527 if(context == NULL)
528 {
529 return FALSE;
530 }
531
532 /* Open the output file */
533 if(strcmp(file, "-") == 0)
534 {
535 context->of = stdout;
536 }
537 else
538 {
539 context->of = fopen(file, "wb");
540 if(!context->of)
541 {
542 fprintf(stderr, "SvgInit: Failed to open output file '%s': %s\n", file, strerror(errno));
543 return FALSE;
544 }
545 }
546
547 /* Set the initial pen state */
548 SvgSetPen(outContext, ADRAW_COL_BLACK);
549 SvgSetBgPen(outContext, ADRAW_COL_WHITE);
550
551 /* Default to small font */
552 SvgSetFontSize(outContext, ADRAW_FONT_SMALL);
553
554 fprintf(context->of, "<!DOCTYPE svg PUBLIC \"-//W3C//DTD SVG 1.1//EN\"\n"
555 " \"http://www.w3.org/Graphics/SVG/1.1/DTD/svg11.dtd\">\n");
556
557 fprintf(context->of, "<svg version=\"1.1\"\n"
558 " width=\"%upx\" height=\"%upx\"\n"
559 " viewBox=\"0 0 %u %u\"\n"
560 " xmlns=\"http://www.w3.org/2000/svg\" shape-rendering=\"crispEdges\"\n"
561 " stroke-width=\"1\" text-rendering=\"geometricPrecision\">\n",
562 w, h, w, h);
563
564 /* Now fill in the function pointers */
565 outContext->line = SvgLine;
566 outContext->dottedLine = SvgDottedLine;
567 outContext->textL = SvgTextL;
568 outContext->textC = SvgTextC;
569 outContext->textR = SvgTextR;
570 outContext->textWidth = SvgTextWidth;
571 outContext->textHeight = SvgTextHeight;
572 outContext->filledRectangle = SvgFilledRectangle;
573 outContext->filledTriangle = SvgFilledTriangle;
574 outContext->filledCircle = SvgFilledCircle;
575 outContext->arc = SvgArc;
576 outContext->dottedArc = SvgDottedArc;
577 outContext->setPen = SvgSetPen;
578 outContext->setBgPen = SvgSetBgPen;
579 outContext->setFontSize = SvgSetFontSize;
580 outContext->close = SvgClose;
581
582 return TRUE;
583}
