49static std::string trim(const std::string& str,
50 const std::string& whitespace = " \t")
51{
52 size_t strBegin = str.find_first_not_of(whitespace);
53 if (strBegin == std::string::npos)
54 return ""; // no content
55
56 size_t strEnd = str.find_last_not_of(whitespace);
57 size_t strRange = strEnd - strBegin + 1;
58
59 return str.substr(strBegin, strRange);
60}

65static std::string reduce(const std::string& str,
66 const std::string& fill = " ",
67 const std::string& whitespace = " \t")
68{
69 // trim first
70 std::string result = trim(str, whitespace);
71
72 // replace sub ranges
73 size_t beginSpace = result.find_first_of(whitespace);
74 while (beginSpace != std::string::npos)
75 {
76 size_t endSpace = result.find_first_not_of(whitespace, beginSpace);
77 size_t range = endSpace - beginSpace;
78
79 result.replace(beginSpace, range, fill);
80
81 size_t newStart = beginSpace + fill.length();
82 beginSpace = result.find_first_of(whitespace, newStart);
83 }
84
85 return result;
86}

89static void addWords(const std::string &s,Xapian::Document &doc,int wfd)
90{
91 std::istringstream iss(s);
92 std::istream_iterator<std::string> begin(iss),end,it;
93 for (it=begin;it!=end;++it)
94 {
95 std::string word = *it;
96 std::string lword = word;
97 std::transform(lword.begin(), lword.end(), lword.begin(), ::tolower);
98 safeAddTerm(word,doc,wfd);
99 if (lword!=word)
100 {
101 safeAddTerm(lword,doc,wfd);
102 }
103 }
104}

107static void addIdentifiers(const std::string &s,Xapian::Document &doc,int wfd)
108{
109 std::regex id_re("[A-Z_a-z][A-Z_a-z0-9]*");
110 auto id_begin = std::sregex_iterator(s.begin(), s.end(), id_re);
111 auto id_end = std::sregex_iterator();
112
113 for (auto i = id_begin; i!=id_end; ++i)
114 {
115 std::smatch match = *i;
116 safeAddTerm(match.str(),doc,wfd);
117 }
118}

121static void replace_all(std::string& str, const std::string& old, const std::string& repl)
122{
123 size_t pos = 0;
124 while ((pos = str.find(old, pos)) != std::string::npos)
125 {
126 str.replace(pos, old.length(), repl);
127 pos += repl.length();
128 }
129}

132static std::string unescapeXmlEntities(const std::string &s)
133{
134 std::string result=s;
135 replace_all(result,"&gt;",">");
136 replace_all(result,"&lt;","<");
137 replace_all(result,"&apos;","'");
138 replace_all(result,"&quot;","\"");
139 replace_all(result,"&amp;","&");
140 return result;
141}

165 enum FieldNames
166 {
167 UnknownField = 0,
168 TypeField = 1,
169 NameField = 2,
170 ArgsField = 3,
171 TagField = 4,
172 UrlField = 5,
173 KeywordField = 6,
174 TextField = 7
175 };
176
178 void startElement(const std::string &name, const XMLHandlers::Attributes &attrib)
179 {
180 m_data="";
181 if (name=="field")
182 {
183 std::string fieldName = XMLHandlers::value(attrib,"name");
184 if (fieldName=="type") m_curFieldName=TypeField;
185 else if (fieldName=="name") m_curFieldName=NameField;
186 else if (fieldName=="args") m_curFieldName=ArgsField;
187 else if (fieldName=="tag") m_curFieldName=TagField;
188 else if (fieldName=="url") m_curFieldName=UrlField;
189 else if (fieldName=="keywords") m_curFieldName=KeywordField;
190 else if (fieldName=="text") m_curFieldName=TextField;
191 else m_curFieldName=UnknownField;
192 }
193 }
194
196 void endElement(const std::string &name)
197 {
198 if (name=="doc") // </doc>
199 {
200 std::string term = m_doc.get_value(NameField);
201 std::string partTerm;
202 size_t pos = term.rfind("::");
203 if (pos!=std::string::npos)
204 {
205 partTerm = term.substr(pos+2);
206 }
207 if (m_doc.get_value(TypeField)=="class" ||
208 m_doc.get_value(TypeField)=="file" ||
209 m_doc.get_value(TypeField)=="namespace") // containers get highest prio
210 {
211 safeAddTerm(term,m_doc,1000);
212 if (!partTerm.empty())
213 {
214 safeAddTerm(partTerm,m_doc,500);
215 }
216 }
217 else // members and others get lower prio
218 {
219 safeAddTerm(m_doc.get_value(NameField),m_doc,100);
220 if (!partTerm.empty())
221 {
222 safeAddTerm(partTerm,m_doc,50);
223 }
224 }
225 m_db.add_document(m_doc);
226 m_doc.clear_values();
227 m_doc.clear_terms();
228 }
229 else if (name=="field" && m_curFieldName!=UnknownField) // </field>
230 {
231 // strip whitespace from m_data
232 m_data = reduce(m_data);
233 // replace XML entities
234 m_data = unescapeXmlEntities(m_data);
235 // add data to the document
236 m_doc.add_value(m_curFieldName,m_data);
237 switch (m_curFieldName)
238 {
239 case TypeField:
240 case NameField:
241 case TagField:
242 case UrlField:
243 // meta data that is not searchable
244 break;
245 case KeywordField:
246 addWords(m_data,m_doc,50);
247 break;
248 case ArgsField:
249 addIdentifiers(m_data,m_doc,10);
250 break;
251 case TextField:
252 addWords(m_data,m_doc,2);
253 break;
254 default:
255 break;
256 }
257 m_data="";
258 m_curFieldName=UnknownField;
259 }
260 // reset m_data
261 }

293{
294 std::ifstream t(fileName);
295 std::string result;
296 t.seekg(0, std::ios::end);

311int main(int argc,const char **argv)
312{
313 if (argc<2)
314 {
315 usage(argv[0]);
316 }
317 std::string outputDir;
318 for (int i=1;i<argc;i++)
319 {
320 if (std::string(argv[i])=="-o")
321 {
322 if (i>=argc-1)
323 {
324 std::cerr << "Error: missing parameter for -o option" << std::endl;
325 usage(argv[0]);
326 }
327 else
328 {
329 i++;
330 outputDir=argv[i];
331 if (!dirExists(outputDir.c_str()))
332 {
333 std::cerr << "Error: specified output directory does not exist!" << std::endl;
334 usage(argv[0]);
335 }
336 }
337 }
338 else if (std::string(argv[i])=="-h" || std::string(argv[i])=="--help")
339 {
340 usage(argv[0],0);
341 }
342 else if (std::string(argv[i])=="-v" || std::string(argv[i])=="--version")
343 {
344 std::cerr << argv[0] << " version: " << getFullVersion() << std::endl;
345 exit(0);
346 }
347 }
348
349 try
350 {
351 if (!outputDir.empty() && outputDir.at(outputDir.length()-1)!=pathSep)
352 {
353 outputDir+=pathSep;
354 }
355 XMLContentHandler contentHandler(outputDir);
356 XMLHandlers handlers;
357 handlers.startElement = [&contentHandler](const std::string &name,const XMLHandlers::Attributes &attrs) { contentHandler.startElement(name,attrs); };
358 handlers.endElement = [&contentHandler](const std::string &name) { contentHandler.endElement(name); };
359 handlers.characters = [&contentHandler](const std::string &chars) { contentHandler.characters(chars); };
360 handlers.error = [&contentHandler](const std::string &fileName,int lineNr,const std::string &msg) { contentHandler.error(fileName,lineNr,msg); };
361 for (int i=1;i<argc;i++)
362 {
363 if (std::string(argv[i])=="-o")
364 {
365 i++;
366 }
367 else
368 {
369 std::cout << "Processing " << argv[i] << "..." << std::endl;
370 std::string inputStr = fileToString(argv[i]);
371 XMLParser parser(handlers);
372 parser.parse(argv[i],inputStr.c_str(),false);
373 }
374 }
375 }
376 catch(const Xapian::Error &e)
377 {
378 std::cerr << "Caught exception: " << e.get_description() << std::endl;
379 }
380 catch(...)
381 {
382 std::cerr << "Caught an unknown exception" << std::endl;
383 }
384
385 return 0;
386}
