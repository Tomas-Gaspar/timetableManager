81def description_is_empty(description):
82 for content in description.content_:
83 if content.getCategory()==MixedContainer.CategoryText:
84 if not content.getValue().isspace():
85 return False # non space-only text
86 elif not content.getCategory()==MixedContainer.CategoryNone:
87 return False # some internal object like a paragraph
88 return True

174def parse_compound(inDirName,baseName,metrics):
175 rootObj = doxmlparser.compound.parse(inDirName+"/"+baseName+".xml",True)
176 for compounddef in rootObj.get_compounddef():
177 kind = compounddef.get_kind()
178 if kind==DoxCompoundKind.CLASS:
179 metrics.numClasses+=1
180 if is_documented(compounddef):
181 metrics.numDocClasses+=1
182 elif kind==DoxCompoundKind.STRUCT:
183 metrics.numStructs+=1
184 elif kind==DoxCompoundKind.UNION:
185 metrics.numUnions+=1
186 elif kind==DoxCompoundKind.INTERFACE:
187 metrics.numInterfaces+=1
188 elif kind==DoxCompoundKind.EXCEPTION:
189 metrics.numExceptions+=1
190 elif kind==DoxCompoundKind.NAMESPACE:
191 metrics.numNamespaces+=1
192 elif kind==DoxCompoundKind.FILE:
193 metrics.numFiles+=1
194 if is_documented(compounddef):
195 metrics.numDocFiles+=1
196 elif kind==DoxCompoundKind.GROUP:
197 metrics.numGroups+=1
198 elif kind==DoxCompoundKind.PAGE:
199 metrics.numPages+=1
