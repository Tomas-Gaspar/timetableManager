142static void writeEscaped(struct ADrawTag *ctx, const char *string)
143{
144 FILE *f = getPsFile(ctx);
145
146 while(*string != '\0')
147 {
148 unsigned int code, bytes;
149
150 switch(*string)
151 {
152 case '(': fprintf(f, "\\("); break;
153 case ')': fprintf(f, "\\)"); break;
154 default:
155 if(Utf8Decode(string, &code, &bytes))
156 {
157 fprintf(f, "\\%o", code);
158 string += bytes - 1;
159 }
160 else
161 {
162 fputc(*string, f);
163 }
164 break;
165 }
166
167 string++;
168 }
169}

171static void setColour(struct ADrawTag *ctx,
172 ADrawColour col)
173{
174 float r, g, b;
175
176 /* Extract RGB values */
177 r = (float)((col & 0xff0000) >> 16);
178 g = (float)((col & 0x00ff00) >> 8);
179 b = (float)((col & 0x0000ff) >> 0);
180
181 /* Normalise */
182 r /= 255.0f;
183 g /= 255.0f;
184 b /= 255.0f;
185
186 /* Generate output command */
187 fprintf(getPsFile(ctx), "%f %f %f setrgbcolor\n", r ,g ,b);
188}

194unsigned int PsTextWidth(struct ADrawTag *ctx,
195 const char *string)
196{
197 unsigned long width = 0;
198
199 while(*string != '\0')
200 {
201 int i = *string & 0xff;
202 unsigned long w = PsHelvetica.widths[i];
203
204 /* Ignore undefined characters */
205 width += w > 0 ? w : 0;
206
207 string++;
208 }
209
210 return getSpace(ctx, width);
211}

266void PsTextR(struct ADrawTag *ctx,
267 unsigned int x,
268 unsigned int y,
269 const char *string)
270{
271 PsContext *context = getPsCtx(ctx);
272
273 /* Push the string and get its width */
274 fprintf(getPsFile(ctx), "(");
275 writeEscaped(ctx, string);
276 fprintf(getPsFile(ctx), ") dup stringwidth\n");
277
278 /* Draw the background box */
279 setColour(ctx, context->penBgColour);
280 fprintf(getPsFile(ctx), "pop " /* Ignore y-value */
281 "dup " /* Duplicate string width */
282 "newpath "
283 "%d %d moveto " /* Bottom left of the box */
284 "0 rlineto " /* Move to bottom right of the box */
285 "0 %d rlineto " /* To top right */
286 "neg 0 rlineto " /* Back to bottom left */
287 "closepath fill\n", /* Done */
288 x, 0-y - getSpace(ctx, PsHelvetica.descender),
289 getSpace(ctx, PsHelvetica.ascender));
290
291 /* Restore pen and show the string */
292 setColour(ctx, context->penColour);
293 fprintf(getPsFile(ctx), "%d %d moveto show\n",
294 x, 0-y - getSpace(ctx, PsHelvetica.descender));
295}

494Boolean PsInit(unsigned int w,
495 unsigned int h,
496 const char *file,
497 struct ADrawTag *outContext)
498{
499 PsContext *context;
500
501 /* Create context */
502 context = outContext->internal = malloc_s(sizeof(PsContext));
503 if(context == NULL)
504 {
505 return FALSE;
506 }
507
508 /* Open the output file */
509 if(strcmp(file, "-") == 0)
510 {
511 context->of = stdout;
512 }
513 else
514 {
515 context->of = fopen(file, "wb");
516 if(!context->of)
517 {
518 fprintf(stderr, "PsInit: Failed to open output file '%s': %s\n", file, strerror(errno));
519 return FALSE;
520 }
521 }
522
523 /* Write the header */
524 fprintf(context->of, "%%!PS-Adobe-3.0 EPSF-2.0\n"
525 "%%%%BoundingBox: 0 0 %.0f %.0f\n", w * PS_OUT_SCALE, h * PS_OUT_SCALE);
526 fprintf(context->of, "%%%%Creator: mscgen %s\n", PACKAGE_VERSION);
527 fprintf(context->of, "%%%%EndComments\n");
528
529 /* Shrink everything by 70% */
530 fprintf(context->of, "%f %f scale\n", PS_OUT_SCALE, PS_OUT_SCALE);
531
532 /* Create clipping rectangle to constrain dimensions */
533 fprintf(context->of, "0 0 moveto\n");
534 fprintf(context->of, "0 %u lineto\n", h);
535 fprintf(context->of, "%u %u lineto\n", w, h);
536 fprintf(context->of, "%u 0 lineto\n", w);
537 fprintf(context->of, "closepath\n");
538 fprintf(context->of, "clip\n");
539 fprintf(context->of, "%%PageTrailer\n");
540 fprintf(context->of, "%%Page: 1 1\n");
541
542 /* Set default font */
543 fprintf(context->of, "/Helvetica findfont\n");
544 fprintf(context->of, "10 scalefont\n");
545 fprintf(context->of, "setfont\n");
546
547 /* Get the default font size */
548 PsSetFontSize(outContext, ADRAW_FONT_SMALL);
549
550 /* Translate up by the height, y-axis will be inverted */
551 fprintf(context->of, "0 %d translate\n", h);
552
553 /* Arc drawing function */
554 fprintf(context->of, "/mtrx matrix def\n"
555 "/ellipse\n"
556 " { /endangle exch def\n"
557 " /startangle exch def\n"
558 " /ydia exch def\n"
559 " /xdia exch def\n"
560 " /y exch def\n"
561 " /x exch def\n"
562 " /savematrix mtrx currentmatrix def\n"
563 " x y translate\n"
564 " xdia 2 div ydia 2 div scale\n"
565 " 1 -1 scale\n"
566 " 0 0 1 startangle endangle arc\n"
567 " savematrix setmatrix\n"
568 "} def\n");
569
570 /* Set the current pen colours */
571 context->penColour = ADRAW_COL_BLACK;
572 context->penBgColour = ADRAW_COL_WHITE;
573
574 /* Now fill in the function pointers */
575 outContext->line = PsLine;
576 outContext->dottedLine = PsDottedLine;
577 outContext->textL = PsTextL;
578 outContext->textC = PsTextC;
579 outContext->textR = PsTextR;
580 outContext->textWidth = PsTextWidth;
581 outContext->textHeight = PsTextHeight;
582 outContext->filledRectangle = PsFilledRectangle;
583 outContext->filledTriangle = PsFilledTriangle;
584 outContext->filledCircle = PsFilledCircle;
585 outContext->arc = PsArc;
586 outContext->dottedArc = PsDottedArc;
587 outContext->setPen = PsSetPen;
588 outContext->setBgPen = PsSetBgPen;
589 outContext->setFontSize = PsSetFontSize;
590 outContext->close = PsClose;
591
592 return TRUE;
593}
