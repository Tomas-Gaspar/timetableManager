33static bool convertMapFile(TextStream &t,const QCString &mapName,const QCString &relPath,
34 const QCString &context)
35{
36 std::ifstream f(mapName.str(),std::ifstream::in);
37 if (!f.is_open())
38 {
39 err("failed to open map file %s for inclusion in the docs!\n"
40 "If you installed Graphviz/dot after a previous failing run, \n"
41 "try deleting the output directory and rerun doxygen.\n",qPrint(mapName));
42 return false;
43 }
44 const int maxLineLen=1024;
45 char url[maxLineLen];
46 char ref[maxLineLen];
47 int x1,y1,x2,y2;
48 std::string line;
49 while (getline(f,line))
50 {
51 bool isRef = false;
52 //printf("ReadLine '%s'\n",line.c_str());
53 if (qstrncmp(line.c_str(),"rect",4)==0)
54 {
55 // obtain the url and the coordinates in the order used by graphviz-1.5
56 sscanf(line.c_str(),"rect %s %d,%d %d,%d",url,&x1,&y1,&x2,&y2);
57
58 if (qstrcmp(url,"\\ref")==0 || qstrcmp(url,"@ref")==0)
59 {
60 isRef = true;
61 sscanf(line.c_str(),"rect %s %s %d,%d %d,%d",ref,url,&x1,&y1,&x2,&y2);
62 }
63
64 // sanity checks
65 if (y2<y1) { int temp=y2; y2=y1; y1=temp; }
66 if (x2<x1) { int temp=x2; x2=x1; x1=temp; }
67
68 t << "<area href=\"";
69
70 if ( isRef )
71 {
72 // handle doxygen \ref tag URL reference
73
74 auto parser { createDocParser() };
75 auto dfAst { createRef( *parser.get(), url, context ) };
76 auto dfAstImpl = dynamic_cast<const DocNodeAST*>(dfAst.get());
77 const DocRef *df = std::get_if<DocRef>(&dfAstImpl->root);
78 t << externalRef(relPath,df->ref(),TRUE);
79 if (!df->file().isEmpty()) t << addHtmlExtensionIfMissing(df->file());
80 if (!df->anchor().isEmpty()) t << "#" << df->anchor();
81 }
82 else
83 {
84 t << url;
85 }
86 t << "\" shape=\"rect\" coords=\""
87 << x1 << "," << y1 << "," << x2 << "," << y2 << "\""
88 << " alt=\"\"/>\n";
89 }
90 }
91
92 return true;
93}

95void writeMscGraphFromFile(const QCString &inFile,const QCString &outDir,
96 const QCString &outFile,MscOutputFormat format,
97 const QCString &srcFile,int srcLine
98 )
99{
100 QCString absOutFile = outDir;
101 absOutFile+=Portable::pathSeparator();
102 absOutFile+=outFile;
103
104 mscgen_format_t msc_format;
105 QCString imgName = absOutFile;
106 switch (format)
107 {
108 case MSC_BITMAP:
109 msc_format = mscgen_format_png;
110 imgName+=".png";
111 break;
112 case MSC_EPS:
113 msc_format = mscgen_format_eps;
114 imgName+=".eps";
115 break;
116 case MSC_SVG:
117 msc_format = mscgen_format_svg;
118 imgName+=".svg";
119 break;
120 default:
121 return;
122 }
123 int code;
124 if ((code=mscgen_generate(inFile.data(),imgName.data(),msc_format))!=0)
125 {
126 err_full(srcFile,srcLine,"Problems generating msc output (error=%s). Look for typos in you msc file %s\n",
127 mscgen_error2str(code),qPrint(inFile));
128 return;
129 }
130
131 if ( (format==MSC_EPS) && (Config_getBool(USE_PDFLATEX)) )
132 {
133 QCString epstopdfArgs(maxCmdLine);
134 epstopdfArgs.sprintf("\"%s.eps\" --outfile=\"%s.pdf\"",
135 qPrint(absOutFile),qPrint(absOutFile));
136 Portable::sysTimerStart();
137 if (Portable::system("epstopdf",epstopdfArgs)!=0)
138 {
139 err_full(srcFile,srcLine,"Problems running epstopdf when processing '%s.eps'. Check your TeX installation!\n",
140 qPrint(absOutFile));
141 }
142 Portable::sysTimerStop();
143 }
144
145 Doxygen::indexList->addImageFile(imgName);
146
147}

149static QCString getMscImageMapFromFile(const QCString& inFile, const QCString& outDir,
150 const QCString& relPath,const QCString& context,
151 bool writeSVGMap,const QCString &srcFile,int srcLine)
152{
153 QCString outFile = inFile + ".map";
154
155 int code;
156 if ((code=mscgen_generate(inFile.data(),outFile.data(),
157 writeSVGMap ? mscgen_format_svgmap : mscgen_format_pngmap))!=0)
158 {
159 err_full(srcFile,srcLine,"Problems generating msc output (error=%s). Look for typos in you msc file %s\n",
160 mscgen_error2str(code),qPrint(inFile));
161 return "";
162 }
163
164 TextStream t;
165 convertMapFile(t, outFile, relPath, context);
166
167 Dir().remove(outFile.str());
168
169 return t.str();
170}

172void writeMscImageMapFromFile(TextStream &t,const QCString &inFile,
173 const QCString &outDir,
174 const QCString &relPath,
175 const QCString &baseName,
176 const QCString &context,
177 MscOutputFormat format,
178 const QCString &srcFile,
179 int srcLine
180 )
181{
182 QCString mapName = baseName+".map";
183 t << "<img src=\"" << relPath << baseName << ".";
184 switch (format)
185 {
186 case MSC_BITMAP:
187 t << "png";
188 break;
189 case MSC_EPS:
190 t << "eps";
191 break;
192 case MSC_SVG:
193 t << "svg";
194 break;
195 default:
196 t << "unknown";
197 }
198 QCString imap = getMscImageMapFromFile(inFile,outDir,relPath,context,format==MSC_SVG,srcFile,srcLine);
199 if (!imap.isEmpty())
200 {
201 t << "\" alt=\""
202 << baseName << "\" border=\"0\" usemap=\"#" << mapName << "\"/>\n";
203 t << "<map name=\"" << mapName << "\" id=\"" << mapName << "\">" << imap << "</map>\n";
204 }
205 else
206 {
207 t << "\" alt=\"" << baseName << "\" border=\"0\"/>\n";
208 }
209}
