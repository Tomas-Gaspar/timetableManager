154 QStringList fileNames = QFileDialog::getOpenFileNames();
155
156 if (!fileNames.isEmpty())
157 {
158 QStringList::Iterator it;
159 for ( it= fileNames.begin(); it != fileNames.end(); ++it )
160 {
161 QString fileName;
162 QDir dir(path);
163 if (!MainWindow::instance().configFileName().isEmpty() && dir.exists())
164 {
165 fileName = dir.relativeFilePath(*it);
166 }
167 if (fileName.isEmpty())
168 {
169 fileName = *it;
170 }
171 m_lb->addItem(fileName);
172 m_strList.append(fileName);
173 m_value = m_strList;
174 updateDefault();
175 emit changed();
176 }
177 m_le->setText(m_strList[0]);
178 }

260bool InputStrList::isDefault()
261{
262 if (m_strList==m_default) return true;
263
264 auto it1 = m_strList.begin();
265 auto it2 = m_default.begin();
266 while (it1!=m_strList.end() && (*it1).isEmpty())
267 {
268 ++it1;
269 }
270 while (it2!=m_default.end() && (*it2).isEmpty())
271 {
272 ++it2;
273 }
274 // both lists are empty
275 if (it1==m_strList.end() && it2==m_default.end()) return true;
276
277 // one list is empty but the other is not
278 if (it1==m_strList.end()) return false;
279 if (it2==m_default.end()) return false;
280
281 it1 = m_strList.begin();
282 it2 = m_default.begin();
283 while (it1!=m_strList.end() && it2!=m_default.end())
284 {
285 // skip over empty values
286 while (it1!=m_strList.end() && (*it1).isEmpty())
287 {
288 ++it1;
289 }
290 while (it2!=m_default.end() && (*it2).isEmpty())
291 {
292 ++it2;
293 }
294 if ((it1!=m_strList.end()) && (it2!=m_default.end()))
295 {
296 if ((*it1).trimmed()!= (*it2).trimmed()) // difference so not the default
297 {
298 return false;
299 }
300 ++it1;
301 ++it2;
302 }
303 else if ((it1!=m_strList.end()) || (it2!=m_default.end()))
304 {
305 // one list empty so cannot be the default
306 return false;
307 }
308 }
309
310 return true;
311}
