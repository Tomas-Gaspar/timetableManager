185static int getArcGradient(Context *ctx, Msc m, const RowInfo *rowInfo, unsigned int row)
186{
187 const char *s = MscGetCurrentArcAttrib(m, MSC_ATTR_ARC_SKIP);
188 unsigned int v = ctx->opts.arcGradient;
189
190 if (s != NULL && rowInfo != NULL)
191 {
192 const unsigned int rowCount = MscGetNumArcs(m) - MscGetNumParallelArcs(m);
193 unsigned int skip;
194
195 if (sscanf(s, "%u", &skip) == 1)
196 {
197 unsigned int ystart = rowInfo[row].arcliney;
198 unsigned int yend = rowInfo[M_Min(rowCount - 1, row + skip)].arcliney;
199
200 v += yend - ystart;
201 }
202 else
203 {
204 fprintf(stderr, "Warning: Non-integer arcskip value: %s\n", s);
205 }
206 }
207
208 return v;
209}

249static char *splitStringToWidth(Context *ctx, char *l, unsigned int width)
250{
251 char *p = l + strlen(l);
252 char *orig = NULL;
253 int m, n;
254
255 if (ctx->drw.textWidth(&ctx->drw, l) > width)
256 {
257 /* Duplicate the original string */
258 orig = strdup_s(l);
259
260 /* Now remove words from the line until it fits the available width */
261 do
262 {
263 /* Skip back 1 word */
264 while (!isspace(*p) && p > l)
265 {
266 p--;
267 }
268
269 if (p > l)
270 {
271 *p = '\0';
272 }
273 }
274 while (ctx->drw.textWidth(&ctx->drw, l) > width && p > l);
275
276 /* Check if the first word is bigger than the available space;
277 * we need to hyphenate in this case.
278 */
279 if (p == l)
280 {
281 const unsigned int hyphenWidth = ctx->drw.textWidth(&ctx->drw, "-");
282
283 /* Find the end of the first word */
284 while (!isspace(*p) && *p != '\0')
285 {
286 p++;
287 }
288
289 /* Start removing characters from the word */
290 do
291 {
292 *p = '\0';
293 p--;
294 }
295 while (ctx->drw.textWidth(&ctx->drw, l) + hyphenWidth > width && p > l);
296
297 /* Add a hyphen */
298 *p = '-';
299 }
300
301 /* Copy the remaining line to the start of the string */
302 m = 0;
303 n = (int)(p - l);
304
305 while (isspace(orig[n]) && orig[n] != '\0')
306 {
307 n++;
308 }
309
310 do
311 {
312 orig[m++] = orig[n++];
313 }
314 while (orig[m - 1] != '\0');
315 }
316
317 return orig;
318}

337static unsigned int computeLabelLines(Context *ctx,
338 Msc m,
339 const MscArcType arcType,
340 char ***lines,
341 const char *label,
342 int startCol,
343 int endCol)
344{
345 unsigned int width;
346 unsigned int nAllocLines = 8;
347 char **retLines = malloc_s(sizeof(char *) * nAllocLines);
348 unsigned int c = 0;
349
350 assert(startCol >= 0 && startCol < (signed)MscGetNumEntities(m));
351 assert(startCol >= -1 && startCol < (signed)MscGetNumEntities(m));
352
353 /* Compute available width for text */
354 if (isBoxArc(arcType) || ctx->opts.wordWrapArcLabels)
355 {
356 if (endCol == -1)
357 {
358 /* This is a special case for a broadcast arc */
359 width = ctx->opts.entitySpacing * MscGetNumEntities(m);
360 }
361 else if (startCol < endCol)
362 {
363 width = ctx->opts.entitySpacing * (1 + (endCol - startCol));
364 }
365 else
366 {
367 width = ctx->opts.entitySpacing * (1 + (startCol - endCol));
368 }
369
370 /* Reduce the width due to the box borders */
371 if (isBoxArc(arcType))
372 {
373 width -= (ctx->opts.boxSpacing + ctx->opts.boxInternalBorder) * 2;
374 }
375
376 if (arcType == MSC_ARC_NOTE)
377 {
378 width -= ctx->opts.noteCorner;
379 }
380 }
381 else
382 {
383 width = UINT_MAX;
384 }
385
386 /* Split the input label into lines */
387 while (label != NULL)
388 {
389 /* First split around user specified lines with literal '\n' */
390 char *nextLine = strstr(label, "\\n");
391 if (nextLine)
392 {
393 const int lineLen = (int)(nextLine - label);
394
395 /* Allocate storage and duplicate the line */
396 retLines[c] = malloc_s(lineLen + 1);
397 memcpy(retLines[c], label, lineLen);
398 retLines[c][lineLen] = '\0';
399
400 /* Advance the label */
401 label = nextLine + 2;
402 }
403 else
404 {
405 /* Duplicate the final line */
406 retLines[c] = strdup_s(label);
407 label = NULL;
408 }
409
410 /* Now split the line as required to wrap into the space available */
411 do
412 {
413 /* Check if more storage maybe needed */
414 if (c + 2 >= nAllocLines)
415 {
416 nAllocLines += 8;
417 retLines = realloc_s(retLines, sizeof(char *) * nAllocLines);
418 }
419
420 retLines[c + 1] = splitStringToWidth(ctx, retLines[c], width);
421 c++;
422 }
423 while (retLines[c] != NULL);
424 }
425
426 /* Return the array of lines and the count */
427 *lines = retLines;
428
429 return c;
430}

443static RowInfo *computeCanvasSize(Context *ctx,
444 Msc m,
445 unsigned int *w,
446 unsigned int *h)
447{
448 const unsigned int rowCount = MscGetNumArcs(m) - MscGetNumParallelArcs(m);
449 const unsigned int textHeight = ctx->drw.textHeight(&ctx->drw);
450 RowInfo *rowHeight;
451 unsigned int nextYmin, ymin, ymax, yskipmax, row;
452
453 /* Allocate storage for the height of each row */
454 rowHeight = zalloc_s(sizeof(RowInfo) * rowCount);
455 row = 0;
456
457 nextYmin = ymin = ctx->opts.entityHeadGap;
458 yskipmax = 0;
459
460 ymax = 0;
461 MscResetArcIterator(m);
462 do
463 {
464 const MscArcType arcType = MscGetCurrentArcType(m);
465 const int arcGradient = isBoxArc(arcType) ? 0 : getArcGradient(ctx, m, NULL, 0);
466 char **arcLabelLines = NULL;
467 unsigned int arcLabelLineCount = 0;
468 int startCol = -1, endCol = -1;
469
470 if (arcType == MSC_ARC_PARALLEL)
471 {
472 assert(row > 0);
473
474 row--;
475
476 ymin = rowHeight[row].ymin;
477 nextYmin = rowHeight[row].ymax;
478 }
479 else
480 {
481 /* Get the entity indices */
482 if (arcType != MSC_ARC_DISCO && arcType != MSC_ARC_DIVIDER && arcType != MSC_ARC_SPACE)
483 {
484 startCol = MscGetEntityIndex(m, MscGetCurrentArcSource(m));
485 endCol = MscGetEntityIndex(m, MscGetCurrentArcDest(m));
486 }
487 else
488 {
489 /* Discontinuity or parallel arc spans whole chart */
490 startCol = 0;
491 endCol = MscGetNumEntities(m) - 1;
492 }
493
494 /* Work out how the label fits the gap between entities */
495 arcLabelLineCount = computeLabelLines(ctx, m, arcType, &arcLabelLines,
496 MscGetCurrentArcAttrib(m, MSC_ATTR_LABEL),
497 startCol, endCol);
498
499 assert(row < rowCount);
500
501 /* Update the max line count for the row */
502 if (arcLabelLineCount > rowHeight[row].maxTextLines)
503 {
504 rowHeight[row].maxTextLines = arcLabelLineCount;
505 }
506
507 freeLabelLines(arcLabelLineCount, arcLabelLines);
508
509 /* Compute the height of this arc */
510 if (arcType != MSC_ARC_DISCO && arcType != MSC_ARC_DIVIDER && arcType != MSC_ARC_SPACE)
511 {
512 ymax = ymin + ctx->opts.arcSpacing;
513 ymax += (M_Max(rowHeight[row].maxTextLines, 2) * textHeight);
514 }
515 else
516 {
517 ymax = ymin + ctx->opts.arcSpacing;
518 ymax += (M_Max(rowHeight[row].maxTextLines, 1) * textHeight);
519 }
520
521 /* Update next potential row start */
522 if (ymax > nextYmin)
523 {
524 nextYmin = ymax;
525 }
526
527 /* Compute the dimensions for the completed row */
528 rowHeight[row].ymin = ymin;
529 rowHeight[row].ymax = nextYmin - ctx->opts.arcSpacing;
530 rowHeight[row].arcliney = rowHeight[row].ymin + (rowHeight[row].ymax - rowHeight[row].ymin) / 2;
531 row++;
532
533 /* Start new row */
534 ymin = nextYmin;
535 }
536
537 /* Keep a track of where the gradient may cause the graph to end */
538 if (ymax + arcGradient > ymax)
539 {
540 yskipmax = ymax + arcGradient;
541 }
542
543 }
544 while (MscNextArc(m));
545
546 if (ymax < yskipmax)
547 ymax = yskipmax;
548
549 /* Set the return values */
550 *w = MscGetNumEntities(m) * ctx->opts.entitySpacing;
551 *h = ymax;
552
553 return rowHeight;
554}

603static char *getLine(const char *string,
604 unsigned int line,
605 char *const out,
606 const unsigned int outLen)
607{
608 const char *lineStart, *lineEnd;
609 unsigned int lineLen;
610
611 /* Setup for the loop */
612 lineEnd = NULL;
613 line++;
614
615 do
616 {
617 /* Check if this is the first or a repeat iteration */
618 if(lineEnd)
619 {
620 lineStart = lineEnd + 2;
621 }
622 else
623 {
624 lineStart = string;
625 }
626
627 /* Search for next delimited */
628 lineEnd = strstr(lineStart, "\\n");
629
630 line--;
631 }
632 while (line > 0 && lineEnd != NULL);
633
634 /* Determine the length of the line */
635 if(lineEnd != NULL)
636 {
637 lineLen = (unsigned int)(lineEnd - lineStart);
638 }
639 else
640 {
641 lineLen = (unsigned int)(strlen(string) - (lineStart - string));
642 }
643
644 /* Clamp the length to the buffer */
645 if(lineLen > outLen - 1)
646 {
647 lineLen = outLen - 1;
648 }
649
650 /* Copy desired characters */
651 memcpy(out, lineStart, lineLen);
652
653 /* NULL terminate */
654 out[lineLen] = '\0';
655
656 return out;
657}

677static void entityText(Context *ctx,
678 FILE *ismap,
679 unsigned int x,
680 unsigned int y,
681 const char *entLabel,
682 const char *entUrl,
683 const char *entId,
684 const char *entIdUrl,
685 const char *entColour,
686 const char *entBgColour)
687{
688 if(entLabel)
689 {
690 const unsigned int lines = countLines(entLabel);
691 unsigned int l;
692 char lineBuffer[1024];
693
694 /* Adjust y to be above the writing line */
695 y -= ctx->drw.textHeight(&ctx->drw) * (lines - 1);
696
697 for (l = 0; l < lines - 1; l++)
698 {
699 char *lineLabel = getLine(entLabel, l, lineBuffer, sizeof(lineBuffer));
700 unsigned int width = ctx->drw.textWidth(&ctx->drw, lineLabel);
701
702 /* Push text down one line */
703 y += ctx->drw.textHeight(&ctx->drw);
704
705 /* Check if a URL is associated */
706 if(entUrl)
707 {
708 /* If no explicit colour has been set, make URLS blue */
709 ctx->drw.setPen(&ctx->drw, ADRAW_COL_BLUE);
710
711 /* Image map output */
712 ismapRect(ismap,
713 entUrl,
714 x - (width / 2), y - ctx->drw.textHeight(&ctx->drw),
715 x + (width / 2), y);
716 }
717
718 /* Set to the explicit colours if directed */
719 if(entColour != NULL)
720 {
721 ctx->drw.setPen(&ctx->drw, ADrawGetColour(entColour));
722 }
723
724 if(entBgColour != NULL)
725 {
726 ctx->drw.setBgPen(&ctx->drw, ADrawGetColour(entBgColour));
727 }
728

730 ctx->drw.textC (&ctx->drw, x, y, lineLabel);
731 ctx->drw.setPen(&ctx->drw, ADRAW_COL_BLACK);
732 ctx->drw.setBgPen(&ctx->drw, ADRAW_COL_WHITE);
733
734 /* Render the Id of the title, if specified and for first line only */
735 if(entId && l == 0)
736 {
737 unsigned int idwidth;
738 int idx, idy;

740 idy = y - ctx->drw.textHeight(&ctx->drw);
741 idx = x + (width / 2);
742

744
745 idwidth = ctx->drw.textWidth(&ctx->drw, entId);
746 idy += (ctx->drw.textHeight(&ctx->drw) + 1) / 2;
747
748 if(entIdUrl)
749 {
750 ctx->drw.setPen(&ctx->drw, ADRAW_COL_BLUE);

752 ctx->drw.setPen(&ctx->drw, ADRAW_COL_BLACK);
753
754 /* Image map output */
755 ismapRect(ismap,
756 entIdUrl,
757 idx, idy - ctx->drw.textHeight(&ctx->drw),
758 idx + idwidth, idy);
759 }
760 else
761 {
762 ctx->drw.textR(&ctx->drw, idx, idy, entId);
763 }
764
765 ctx->drw.setFontSize(&ctx->drw, ADRAW_FONT_SMALL);
766 }
767 }
768 }

904static void arcBox(Context *ctx,
905 unsigned int ymin,
906 unsigned int ymax,
907 unsigned int boxStart,
908 unsigned int boxEnd,
909 MscArcType boxType,
910 const char *lineColour,
911 const char *bgColour)
912{
913 unsigned int t;
914
915 /* Ensure the start is less than or equal to the end */
916 if(boxStart > boxEnd)
917 {
918 t = boxEnd;
919 boxEnd = boxStart;
920 boxStart = t;
921 }
922
923 /* Now draw the box */
924 unsigned int x1 = (ctx->opts.entitySpacing * boxStart) + ctx->opts.boxSpacing;
925 unsigned int x2 = ctx->opts.entitySpacing * (boxEnd + 1) - ctx->opts.boxSpacing;
926 unsigned int ymid = (ymin + ymax) / 2;
927
928 /* Set colour for the background area */
929 if(bgColour != NULL)
930 {
931 ctx->drw.setPen(&ctx->drw, ADrawGetColour(bgColour));
932 }
933 else
934 {
935 ctx->drw.setPen(&ctx->drw, ADRAW_COL_WHITE);
936 }
937
938 /* Draw the background to overwrite the entity lines */
939 switch(boxType)
940 {
941 case MSC_ARC_BOX:
942 ctx->drw.filledRectangle(&ctx->drw, x1, ymin, x2, ymax);
943 break;
944
945 case MSC_ARC_RBOX:
946 ctx->drw.filledRectangle(&ctx->drw, x1 + ctx->opts.rboxArc, ymin, x2 - ctx->opts.rboxArc, ymax);
947 ctx->drw.filledRectangle(&ctx->drw, x1, ymin + ctx->opts.rboxArc, x2, ymax - ctx->opts.rboxArc);
948 ctx->drw.filledCircle(&ctx->drw, x1 + ctx->opts.rboxArc, ymin + ctx->opts.rboxArc, ctx->opts.rboxArc);
949 ctx->drw.filledCircle(&ctx->drw, x2 - ctx->opts.rboxArc, ymin + ctx->opts.rboxArc, ctx->opts.rboxArc);
950 ctx->drw.filledCircle(&ctx->drw, x1 + ctx->opts.rboxArc, ymax - ctx->opts.rboxArc, ctx->opts.rboxArc);
951 ctx->drw.filledCircle(&ctx->drw, x2 - ctx->opts.rboxArc, ymax - ctx->opts.rboxArc, ctx->opts.rboxArc);
952 break;
953
954 case MSC_ARC_NOTE:
955 ctx->drw.filledRectangle(&ctx->drw, x1, ymin, x2 - ctx->opts.noteCorner, ymax);
956 ctx->drw.filledRectangle(&ctx->drw, x1, ymin + ctx->opts.noteCorner, x2, ymax);
957 ctx->drw.filledTriangle(&ctx->drw, x2 - ctx->opts.noteCorner, ymin,
958 x2, ymin + ctx->opts.noteCorner,
959 x2 - ctx->opts.noteCorner, ymin + ctx->opts.noteCorner);
960 break;
961
962 case MSC_ARC_ABOX:
963 ctx->drw.filledRectangle(&ctx->drw, x1 + ctx->opts.aboxSlope, ymin, x2 - ctx->opts.aboxSlope, ymax);
964 ctx->drw.filledTriangle(&ctx->drw, x1 + ctx->opts.aboxSlope, ymin,
965 x1 + ctx->opts.aboxSlope, ymax,
966 x1, ymid);
967 ctx->drw.filledTriangle(&ctx->drw, x2 - ctx->opts.aboxSlope, ymin,
968 x2 - ctx->opts.aboxSlope, ymax,
969 x2, ymid);
970 break;
971
972 default:
973 assert(0);
974 }
975
976 /* Setup the colour for rendering the boxes */
977 if(lineColour)
978 {
979 ctx->drw.setPen(&ctx->drw, ADrawGetColour(lineColour));
980 }
981 else
982 {
983 ctx->drw.setPen(&ctx->drw, ADRAW_COL_BLACK);
984 }
985
986 /* Draw the outline */
987 switch(boxType)
988 {
989 case MSC_ARC_BOX:
990 ctx->drw.line(&ctx->drw, x1, ymin, x2, ymin);
991 ctx->drw.line(&ctx->drw, x1, ymax, x2, ymax);
992 ctx->drw.line(&ctx->drw, x1, ymin, x1, ymax);
993 ctx->drw.line(&ctx->drw, x2, ymin, x2, ymax);
994 break;
995
996 case MSC_ARC_NOTE:
997 ctx->drw.line(&ctx->drw, x1, ymin, x2 - ctx->opts.noteCorner, ymin);
998 ctx->drw.line(&ctx->drw, x1, ymax, x2, ymax);
999 ctx->drw.line(&ctx->drw, x1, ymin, x1, ymax);
1000 ctx->drw.line(&ctx->drw, x2, ymin + ctx->opts.noteCorner, x2, ymax);
1001 ctx->drw.line(&ctx->drw, x2 - ctx->opts.noteCorner, ymin,
1002 x2, ymin + ctx->opts.noteCorner);
1003 ctx->drw.line(&ctx->drw, x2 - ctx->opts.noteCorner, ymin,
1004 x2 - ctx->opts.noteCorner, ymin + ctx->opts.noteCorner);
1005 ctx->drw.line(&ctx->drw, x2, ymin + ctx->opts.noteCorner,
1006 x2 - ctx->opts.noteCorner, ymin + ctx->opts.noteCorner);
1007 break;
1008
1009 case MSC_ARC_RBOX:
1010 ctx->drw.line(&ctx->drw, x1 + ctx->opts.rboxArc, ymin, x2 - ctx->opts.rboxArc, ymin);
1011 ctx->drw.line(&ctx->drw, x1 + ctx->opts.rboxArc, ymax, x2 - ctx->opts.rboxArc, ymax);
1012 ctx->drw.line(&ctx->drw, x1, ymin + ctx->opts.rboxArc, x1, ymax - ctx->opts.rboxArc);
1013 ctx->drw.line(&ctx->drw, x2, ymin + ctx->opts.rboxArc, x2, ymax - ctx->opts.rboxArc);
1014
1015 ctx->drw.arc(&ctx->drw, x1 + ctx->opts.rboxArc,
1016 ymin + ctx->opts.rboxArc, ctx->opts.rboxArc * 2, ctx->opts.rboxArc * 2,
1017 180, 270);
1018 ctx->drw.arc(&ctx->drw, x2 - ctx->opts.rboxArc,
1019 ymin + ctx->opts.rboxArc, ctx->opts.rboxArc * 2, ctx->opts.rboxArc * 2,
1020 270, 0);
1021 ctx->drw.arc(&ctx->drw, x2 - ctx->opts.rboxArc,
1022 ymax - ctx->opts.rboxArc, ctx->opts.rboxArc * 2, ctx->opts.rboxArc * 2,
1023 0, 90);
1024 ctx->drw.arc(&ctx->drw, x1 + ctx->opts.rboxArc,
1025 ymax - ctx->opts.rboxArc, ctx->opts.rboxArc * 2, ctx->opts.rboxArc * 2,
1026 90, 180);
1027 break;
1028
1029 case MSC_ARC_ABOX:
1030 ctx->drw.line(&ctx->drw, x1 + ctx->opts.aboxSlope, ymin, x2 - ctx->opts.aboxSlope, ymin);
1031 ctx->drw.line(&ctx->drw, x1 + ctx->opts.aboxSlope, ymax, x2 - ctx->opts.aboxSlope, ymax);
1032 ctx->drw.line(&ctx->drw, x1 + ctx->opts.aboxSlope, ymin, x1, ymid);
1033 ctx->drw.line(&ctx->drw, x1, ymid, x1 + ctx->opts.aboxSlope, ymax);
1034 ctx->drw.line(&ctx->drw, x2 - ctx->opts.aboxSlope, ymin, x2, ymid);
1035 ctx->drw.line(&ctx->drw, x2, ymid, x2 - ctx->opts.aboxSlope, ymax);
1036 break;
1037
1038 default:
1039 assert(0);
1040 }
1041
1042 /* Restore the pen colour if needed */
1043 if(lineColour)
1044 {
1045 ctx->drw.setPen(&ctx->drw, ADRAW_COL_BLACK);
1046 }

1069static void arcText(Context *ctx,
1070 Msc m,
1071 FILE *ismap,
1072 unsigned int outwidth,
1073 unsigned int ymid,
1074 int ygradient,
1075 unsigned int startCol,
1076 unsigned int endCol,
1077 const unsigned int arcLabelLineCount,
1078 char **arcLabelLines,
1079 const char *arcUrl,
1080 const char *arcId,
1081 const char *arcIdUrl,
1082 const char *arcTextColour,
1083 const char *arcTextBgColour,
1084 const MscArcType arcType)
1085{
1086 unsigned int l;
1087 unsigned int y;
1088
1089 /* A single line of normal text is above the midline */
1090 if(arcLabelLineCount == 1 && !isBoxArc(arcType) &&
1091 arcType != MSC_ARC_DISCO && arcType != MSC_ARC_DIVIDER &&
1092 arcType != MSC_ARC_SPACE)
1093 {
1094 y = ymid + (ygradient / 2) - ctx->drw.textHeight(&ctx->drw);
1095 }
1096 else /* Text is vertically centered on the midline */
1097 {
1098 int yoff = ygradient - (ctx->drw.textHeight(&ctx->drw) * arcLabelLineCount);
1099 y = ymid + (yoff / 2);
1100 }
1101
1102 for (l = 0; l < arcLabelLineCount; l++)
1103 {
1104 const char *lineLabel = arcLabelLines[l];
1105 unsigned int width = ctx->drw.textWidth(&ctx->drw, lineLabel);
1106 int x = ((startCol + endCol + 1) * ctx->opts.entitySpacing) / 2;
1107
1108 y += ctx->drw.textHeight(&ctx->drw);
1109
1110 if(startCol != endCol || isBoxArc(arcType))
1111 {
1112 /* Produce central aligned text */
1113 x -= width / 2;
1114 }
1115 else if(startCol < (MscGetNumEntities(m) / 2))
1116 {
1117 /* Form text to the right */
1118 x += ctx->opts.textHGapPre;
1119 }
1120 else
1121 {
1122 /* Form text to the left */
1123 x -= width + ctx->opts.textHGapPost;
1124 }
1125
1126 /* Clip against edges of image */
1127 if(x + width > outwidth)
1128 {
1129 x = outwidth - width;
1130 }
1131
1132 if(x < 0)
1133 {
1134 x = 0;
1135 }
1136
1137 /* Check if a URL is associated */
1138 if(arcUrl)
1139 {
1140 /* Default to blue */
1141 ctx->drw.setPen(&ctx->drw, ADRAW_COL_BLUE);
1142
1143 /* Image map output */
1144 ismapRect(ismap,
1145 arcUrl,
1146 x, y - ctx->drw.textHeight(&ctx->drw),
1147 x + width, y);
1148 }
1149
1150
1151 /* Set to the explicit colours if directed */
1152 if(arcTextColour != NULL)
1153 {
1154 ctx->drw.setPen(&ctx->drw, ADrawGetColour(arcTextColour));
1155 }
1156
1157 if(arcTextBgColour != NULL)
1158 {
1159 ctx->drw.setBgPen(&ctx->drw, ADrawGetColour(arcTextBgColour));
1160 }
1161
1162 /* Render text and restore pen */
1163 ctx->drw.textR (&ctx->drw, x, y, lineLabel);
1164 ctx->drw.setPen(&ctx->drw, ADRAW_COL_BLACK);
1165 ctx->drw.setBgPen(&ctx->drw, ADRAW_COL_WHITE);
1166
1167 /* Render the Id of the arc, if specified and for the first line*/
1168 if(arcId && l == 0)
1169 {
1170 unsigned int idwidth;
1171 int idx, idy;
1172
1173 idy = y - ctx->drw.textHeight(&ctx->drw);
1174 idx = x + width;
1175
1176 ctx->drw.setFontSize(&ctx->drw, ADRAW_FONT_TINY);
1177
1178 idwidth = ctx->drw.textWidth(&ctx->drw, arcId);
1179 idy += (ctx->drw.textHeight(&ctx->drw) + 1) / 2;
1180
1181 if(arcIdUrl)
1182 {
1183 ctx->drw.setPen(&ctx->drw, ADRAW_COL_BLUE);
1184
1185 /* Image map output */
1186 ismapRect(ismap,
1187 arcIdUrl,
1188 idx, idy - ctx->drw.textHeight(&ctx->drw),
1189 idx + idwidth, idy);
1190 }
1191
1192 /* Render text and restore pen and font */
1193 ctx->drw.textR (&ctx->drw, idx, idy, arcId);
1194 ctx->drw.setPen(&ctx->drw, ADRAW_COL_BLACK);
1195 ctx->drw.setFontSize(&ctx->drw, ADRAW_FONT_SMALL);
1196 }
1197 }
1198}

1430/* Perform post-parsing validation of the MSC.
1431 * This checks the passed MSC for various rules which can't easily be tested
1432 * at parse time.
1433 */
1434static Boolean checkMsc(Msc m)
1435{
1436 /* Check all arc entities are known */
1437 MscResetArcIterator(m);
1438 do
1439 {
1440 const MscArcType arcType = MscGetCurrentArcType(m);
1441
1442 if (arcType != MSC_ARC_PARALLEL && arcType != MSC_ARC_DISCO &&
1443 arcType != MSC_ARC_DIVIDER && arcType != MSC_ARC_SPACE)
1444 {
1445 const char *src = MscGetCurrentArcSource(m);
1446 const char *dst = MscGetCurrentArcDest(m);
1447 const int startCol = MscGetEntityIndex(m, src);
1448 const int endCol = MscGetEntityIndex(m, dst);
1449
1450 /* Check the start column is valid */
1451 if (startCol == -1)
1452 {
1453 fprintf(stderr, "Error detected at line %u: Unknown source entity '%s'.\n",
1454 MscGetCurrentArcInputLine(m), src);
1455 return FALSE;
1456 }
1457
1458 if (endCol == -1 && !isBroadcastArc(dst))
1459 {
1460 fprintf(stderr, "Error detected at line %u: Unknown destination entity '%s'.\n",
1461 MscGetCurrentArcInputLine(m), dst);
1462 return FALSE;
1463 }
1464 }
1465 }
1466 while (MscNextArc(m));
1467
1468 return TRUE;
1469}

1472int mscgen_generate(const char *inputFile,
1473 const char *outputFile,
1474 mscgen_format_t format
1475 )
1476{
1477 //printf("mscgen_generate(in=%s,out=%s,format=%d)\n",inputFile,outputFile,format);
1478 ADrawOutputType outType;
1479 ADrawColour *entColourRef = NULL;
1480 const char *outImage = outputFile;
1481 Msc m;
1482 float f;
1483 unsigned int w, h, row, col;
1484 RowInfo *rowInfo = NULL;
1485 Boolean addLines;
1486 FILE *ismap = NULL;
1487#if defined(_WIN32) && !defined(__CYGWIN__)
1488 const char *nullFile = "nul";
1489#else
1490 const char *nullFile = "/dev/null";
1491#endif
1492 Context ctx;
1493 ctx.opts = gDefaultOpts;
1494
1495 switch (format)
1496 {
1497 case mscgen_format_png:
1498 outType = ADRAW_FMT_PNG;
1499 break;
1500 case mscgen_format_eps:
1501 outType = ADRAW_FMT_EPS;
1502 break;
1503 case mscgen_format_svg:
1504 outType = ADRAW_FMT_SVG;
1505 break;
1506 case mscgen_format_pngmap:
1507 outType = ADRAW_FMT_PNG;
1508 outImage = nullFile;
1509 break;
1510 case mscgen_format_svgmap:
1511 outType = ADRAW_FMT_SVG;
1512 outImage = nullFile;
1513 break;
1514 }
1515 /* open file */
1516 FILE *in = fopen(inputFile, "r");
1517 if (!in)
1518 {
1519 fprintf(stderr, "Failed to open input file '%s'\n", inputFile);
1520 return MSCGEN_FILE_ERROR;
1521 }
1522 /* parse file */
1523 m = MscParse(in);
1524 fclose(in);
1525 if (!m || !checkMsc(m))
1526 {
1527 fprintf(stderr, "Input format error for '%s'\n", inputFile);
1528 return MSCGEN_INPUT_FORMAT_ERROR;
1529 }
1530
1531 /* Check if an ismap file should also be generated */
1532 if (format==mscgen_format_pngmap || format==mscgen_format_svgmap)
1533 {
1534 ismap = fopen(outputFile, "w");
1535 if (!ismap)
1536 {
1537 fprintf(stderr, "Failed to open output file '%s': %s\n", outputFile, strerror(errno));
1538 MscFree(m);
1539 return MSCGEN_FILE_ERROR;
1540 }
1541 }
1542
1543 /* Open the drawing context with dummy dimensions */
1544 if (!ADrawOpen(10, 10, nullFile, "", outType, &ctx.drw))
1545 {
1546 if (ismap)
1547 {
1548 fclose(ismap);
1549 }
1550 MscFree(m);
1551 fprintf(stderr, "Failed to open temporary drawing context\n");
1552 return MSCGEN_OUTPUT_CONTEXT_ERROR;
1553 }
1554
1555 /* Now compute ideal canvas size, which may use text metrics */
1556 if (MscGetOptAsFloat(m, MSC_OPT_WIDTH, &f))
1557 {
1558 ctx.opts.idealCanvasWidth = (unsigned int)f;
1559 }
1560 else if (MscGetOptAsFloat(m, MSC_OPT_HSCALE, &f))
1561 {
1562 ctx.opts.idealCanvasWidth *= (unsigned int)f;
1563 }
1564
1565 /* Set the arc gradient if needed */
1566 if (MscGetOptAsFloat(m, MSC_OPT_ARCGRADIENT, &f))
1567 {
1568 ctx.opts.arcGradient = (int)f;
1569 ctx.opts.arcSpacing += ctx.opts.arcGradient;
1570 }
1571
1572 /* Check if word wrapping on arcs other than boxes should be used */
1573 MscGetOptAsBoolean(m, MSC_OPT_WORDWRAPARCS, &ctx.opts.wordWrapArcLabels);
1574
1575 /* Work out the entitySpacing */
1576 if (ctx.opts.idealCanvasWidth / MscGetNumEntities(m) > ctx.opts.entitySpacing)
1577 {
1578 ctx.opts.entitySpacing = ctx.opts.idealCanvasWidth / MscGetNumEntities(m);
1579 }
1580
1581 /* Work out the entityHeadGap */
1582 MscResetEntityIterator(m);
1583 for (col = 0; col < MscGetNumEntities(m); col++)
1584 {
1585 unsigned int lines = countLines(MscGetCurrentEntAttrib(m, MSC_ATTR_LABEL));
1586 unsigned int gap;
1587
1588 /* Get the required gap */
1589 gap = lines * ctx.drw.textHeight(&ctx.drw);
1590 if (gap > ctx.opts.entityHeadGap)
1591 {
1592 ctx.opts.entityHeadGap = gap;
1593 }
1594
1595 MscNextEntity(m);
1596 }
1597
1598 //printf("opts: idealCanvasWidth=%d entitySpacing=%d entityHeadGap=%d "
1599 // "arcSpacing=%d arcGradient=%d boxSpacing=%d boxInternalBorder=%d "
1600 // "rboxArc=%d noteCorner=%d aboxSlope=%d wordWrapArcLabels=%d arrowWidth=%d "
1601 // "arrowHeight=%d loopArcHeight=%d textHGapPre=%d textHGapPost=%d\n",
1602 // ctx.opts.idealCanvasWidth,ctx.opts.entitySpacing,ctx.opts.entityHeadGap,
1603 // ctx.opts.arcSpacing,ctx.opts.arcGradient,ctx.opts.boxSpacing,ctx.opts.boxInternalBorder,
1604 // ctx.opts.rboxArc,ctx.opts.noteCorner,ctx.opts.aboxSlope,ctx.opts.wordWrapArcLabels,ctx.opts.arrowWidth,
1605 // ctx.opts.arrowHeight,ctx.opts.loopArcHeight,ctx.opts.textHGapPre,ctx.opts.textHGapPost);
1606
1607 /* Work out the width and height of the canvas */
1608 rowInfo = computeCanvasSize(&ctx, m, &w , &h);
1609
1610 /* Close the temporary output file */
1611 ctx.drw.close(&ctx.drw);
1612
1613 //printf("opening canvas %d x %d for %s type=%d\n",w,h,outImage,outType);
1614
1615 /* Open the output */
1616 if (!ADrawOpen(w, h, outImage, "", outType, &ctx.drw))
1617 {
1618 free(rowInfo);
1619 if (ismap)
1620 {
1621 fclose(ismap);
1622 }
1623 MscFree(m);
1624 return MSCGEN_OUTPUT_CONTEXT_ERROR;
1625 }
1626
1627 /* Allocate storage for entity heading colours */
1628 entColourRef = malloc_s(MscGetNumEntities(m) * sizeof(ADrawColour));
1629
1630 /* Draw the entity headings */
1631 MscResetEntityIterator(m);
1632 for (col = 0; col < MscGetNumEntities(m); col++)
1633 {
1634 unsigned int x = (ctx.opts.entitySpacing / 2) + (ctx.opts.entitySpacing * col);
1635 const char *line;
1636
1637 /* Titles */
1638 entityText(&ctx, ismap,
1639 x,
1640 ctx.opts.entityHeadGap - (ctx.drw.textHeight(&ctx.drw) / 2),
1641 MscGetCurrentEntAttrib(m, MSC_ATTR_LABEL),
1642 MscGetCurrentEntAttrib(m, MSC_ATTR_URL),
1643 MscGetCurrentEntAttrib(m, MSC_ATTR_ID),
1644 MscGetCurrentEntAttrib(m, MSC_ATTR_IDURL),
1645 MscGetCurrentEntAttrib(m, MSC_ATTR_TEXT_COLOUR),
1646 MscGetCurrentEntAttrib(m, MSC_ATTR_TEXT_BGCOLOUR));
1647
1648 /* Get the colours */
1649 line = MscGetCurrentEntAttrib(m, MSC_ATTR_LINE_COLOUR);
1650 if (line != NULL)
1651 {
1652 entColourRef[col] = ADrawGetColour(line);
1653 }
1654 else
1655 {
1656 entColourRef[col] = ADRAW_COL_BLACK;
1657 }
1658
1659 MscNextEntity(m);
1660 }
1661
1662 /* Draw the arcs */
1663 addLines = TRUE;
1664 row = 0;
1665
1666 MscResetArcIterator(m);
1667 do
1668 {
1669 const MscArcType arcType = MscGetCurrentArcType(m);
1670 const char *arcUrl = MscGetCurrentArcAttrib(m, MSC_ATTR_URL);
1671 const char *arcId = MscGetCurrentArcAttrib(m, MSC_ATTR_ID);

1724 }

1729 }

1731 }

1734 /* Discontinuity or parallel arc spans whole chart */

1736 endCol = MscGetNumEntities(m) - 1;

1761 t, arcLineColour, arcHasArrows,
1762 arcHasBiArrows, arcType);
1763 }
1764 }
1765
1766 /* Fix up the start/end columns to span chart */
1767 startCol = 0;
1768 endCol = MscGetNumEntities(m) - 1;
1769 }

1771 {
1772 /* Check if it is a box, discontinuity arc etc... */

1774 {
1775 if (addLines)
1776 {

1778 }

1780 }

1786 }
1787 }

1789 {
1790 if (addLines)

1793 }

1799

1803 }

1808 (MscGetNumEntities(m) * ctx.opts.entitySpacing) - margin, ymid);
1809
1810 if (arcLineColour != NULL)
1811 {
1812 ctx.drw.setPen(&ctx.drw, ADRAW_COL_BLACK);
1813 }
1814 }
1815 }
1816 else
1817 {
1818 if (addLines)
1819 {
1820 entityLines(&ctx, m, ymin, ymax + ctx.opts.arcSpacing, FALSE, entColourRef);
1821 }
1822 arcLine(&ctx, m, ymid, arcGradient, startCol, endCol, arcLineColour,
1823 arcHasArrows, arcHasBiArrows, arcType);
1824 }
1825 }
1826
1827 /* All may have text */
1828 if (arcLabelLineCount > 0)
1829 {
1830 arcText(&ctx, m, ismap, w, ymid, arcGradient,
1831 startCol, endCol,
1832 arcLabelLineCount, arcLabelLines,
1833 arcUrl, arcId, arcIdUrl,
1834 arcTextColour, arcTextBgColour, arcType);
1835 }
1836
1837 freeLabelLines(arcLabelLineCount, arcLabelLines);
1838
1839 /* Advance the row */
1840 row++;
1841 addLines = TRUE;
1842 }
1843 } while (MscNextArc(m));
1844
1845 /* Skip arcs may require the entity lines to be extended */
1846 entityLines(&ctx, m,
1847 rowInfo[(MscGetNumArcs(m) - MscGetNumParallelArcs(m)) - 1].ymax,
1848 h, FALSE, entColourRef);
1849
1850 if (ismap)
1851 {
1852 fclose(ismap);
1853 }
1854
1855 free(entColourRef);
1856 free(rowInfo);
1857 MscFree(m);
1858 /* Close the context */
1859 ctx.drw.close(&ctx.drw);
1860
1861 return MSCGEN_OK;
1862}
