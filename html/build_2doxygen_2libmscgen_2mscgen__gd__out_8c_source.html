126static int getColourRef(GdoContext *context, ADrawColour col)
127{
128 int t;
129
130 /* Check if the colour is already allocated */
131 for(t = 0; t < context->colourCount; t++)
132 {
133 if(context->colour[t].col == col)
134 {
135 return context->colour[t].ref;
136 }
137 }
138
139 /* Allocate a new colour if there is space */
140 if(t < MAX_COLOURS)
141 {
142 /* Store the colour and allocate a reference */
143 context->colour[t].col = col;
144 context->colour[t].ref = gdImageColorAllocate(context->img,
145 (col & 0xff0000) >> 16,
146 (col & 0x00ff00) >> 8,
147 (col & 0x0000ff) >> 0);
148 context->colourCount++;
149
150 /* Return the new colour reference */
151 return context->colour[t].ref;
152 }
153 else
154 {
155 /* Cannot allocate more colours, so return black by default */
156 return getColourRef(context, ADRAW_COL_BLACK);
157 }
158}

179unsigned int gdoTextWidth(struct ADrawTag *ctx,
180 const char *string)
181{
182#ifndef USE_FREETYPE
183 const unsigned int l = (unsigned int)strlen(string);
184
185 /* Remove 1 pixel since there is usually an uneven gap at
186 * the right of the last character for the fixed width
187 * font.
188 */
189 return l == 0 ? 0 : (getGdoCtx(ctx)->font->w * l) - 1;
190#else
191 GdoContext *context = getGdoCtx(ctx);
192 int rect[8] = { 0, 0, 0, 0, 0, 0, 0, 0 };
193 const char *r;
194
195 r = gdImageStringFT(NULL,
196 rect,
197 context->pen,
198 (char *)context->fontName,
199 context->fontPoints,
200 0,
201 0, 0,
202 (char *)string);
203 if(r)
204 {
205 fprintf(stderr, "Error: gdoTextWidth: %s (GDFONTPATH=%s)\n", r, mscgen_getenv_s("GDFONTPATH"));
206 exit(EXIT_FAILURE);
207 }
208
209 /* Remove 1 pixel since there is usually an uneven gap at
210 * the right of the last character for the fixed width
211 * font.
212 */
213 return rect[2] - 1;
214#endif
215}

246void gdoLine(struct ADrawTag *ctx,
247 unsigned int x1,
248 unsigned int y1,
249 unsigned int x2,
250 unsigned int y2)
251{
252 /* Range check since gdImageLine() takes signed values */
253 if(x1 <= INT_MAX && y1 <= INT_MAX && x2 <= INT_MAX && y2 <= INT_MAX)
254 {
255 /* Anti-aliasing fails if drawing 'backwards' for some octants */
256 if(x1 > x2 && abs((int)x1 - (int)x2) > abs((int)y1 - (int)y2))
257 {
258 swap(&x1, &x2);
259 swap(&y1, &y2);
260 }
261
262 gdImageSetAntiAliased(getGdoImg(ctx), getGdoPen(ctx));
263 gdImageLine(getGdoImg(ctx),
264 x1, y1, x2, y2, gdAntiAliased);
265 }
266}

285void gdoTextR(struct ADrawTag *ctx,
286 unsigned int x,
287 unsigned int y,
288 const char *string)
289{
290 GdoContext *context = getGdoCtx(ctx);
291#ifdef USE_FREETYPE
292 int rect[8] = { 0, 0, 0, 0, 0, 0, 0, 0 };
293 const char *r;
294#endif
295 int textWidth;
296
297 textWidth = gdoTextWidth(ctx, string);
298
299 /* Range check since gdImageFilledRectangle() takes signed values */
300 if(x + textWidth <= INT_MAX && y <= INT_MAX)
301 {
302 gdImageFilledRectangle(getGdoImg(ctx),
303 x,
304 y - (gdoTextHeight(ctx) - 2),
305 x + textWidth,
306 y - 2,
307 context->bgpen);
308
309#ifdef USE_FREETYPE
310 r = gdImageStringFT(getGdoImg(ctx),
311 rect,
312 context->pen,
313 (char *)context->fontName,
314 context->fontPoints,
315 0,
316 x, y - 2,
317 (char *)string);
318
319 if(r)
320 {
321 fprintf(stderr, "Error: gdoTextR: %s (GDFONTPATH=%s)\n", r, mscgen_getenv_s("GDFONTPATH"));
322 exit(EXIT_FAILURE);
323 }
324#else
325 gdImageString(getGdoImg(ctx),
326 getGdoCtx(ctx)->font,
327 x,
328 y - gdoTextHeight(ctx),
329 (unsigned char *)string,
330 getGdoPen(ctx));
331#endif
332 }

495Boolean gdoClose(struct ADrawTag *ctx)
496{
497 GdoContext *context = getGdoCtx(ctx);
498
499 /* Output the image to the disk file in PNG format. */
500 gdImagePng(getGdoImg(ctx), context->outFile);
501 if(context->outFile != stdout)
502 {
503 fclose(context->outFile);
504 }
505
506 /* Destroy the image in memory */
507 gdImageDestroy(context->img);
508
509 /* Free and destroy context */
510 free(context);
511 ctx->internal = NULL;
512
513 return TRUE;
514}

518Boolean GdoInit(unsigned int w,
519 unsigned int h,
520 const char *file,
521 const char *fontName UNUSED,
522 struct ADrawTag *outContext)
523{
524 GdoContext *context;
525
526 /* Range check the size */
527 if(w > INT_MAX || h > INT_MAX)
528 {
529 fprintf(stderr, "Warning: The output image size larger than can be supported for png; output\n"
530 " will be clipped.\n");
531 }
532
533 /* Clip image size to limits */
534 if(w > INT_MAX) w = INT_MAX;
535 if(h > INT_MAX) h = INT_MAX;
536
537 /* Create context */
538 context = outContext->internal = zalloc_s(sizeof(GdoContext));
539 if(context == NULL)
540 {
541 return FALSE;
542 }
543
544 /* Open the output file */
545 if(strcmp(file, "-") == 0)
546 {
547 context->outFile = stdout;
548 }
549 else
550 {
551 context->outFile = fopen(file, "wb");
552 if(!context->outFile)
553 {
554 fprintf(stderr, "GdoInit: Failed to open output file '%s': %s\n", file, strerror(errno));
555 return FALSE;
556 }
557 }
558
559#ifdef USE_FREETYPE
560 /* Request that we use font config strings and store font name */
561 gdFTUseFontConfig(1);
562 context->fontName = fontName;
563
564 assert(fontName != NULL);
565#endif
566
567 /* Allocate the image */
568 context->img = gdImageCreateTrueColor(w, h);
569
570 /* Allocate first colour and clear background */
571 gdImageFilledRectangle(context->img,
572 0, 0,
573 w, h,
574 getColourRef(context, ADRAW_COL_WHITE));
575
576 /* Set pen colour to black and background to white */
577 context->pen = getColourRef(context, ADRAW_COL_BLACK);
578 context->bgpen = getColourRef(context, ADRAW_COL_WHITE);
579
580 /* Get the default font size */
581 gdoSetFontSize(outContext, ADRAW_FONT_SMALL);
582
583 /* Now fill in the function pointers */
584 outContext->line = gdoLine;
585 outContext->dottedLine = gdoDottedLine;
586 outContext->textL = gdoTextL;
587 outContext->textC = gdoTextC;
588 outContext->textR = gdoTextR;
589 outContext->textWidth = gdoTextWidth;
590 outContext->textHeight = gdoTextHeight;
591 outContext->filledRectangle = gdoFilledRectangle;
592 outContext->filledTriangle = gdoFilledTriangle;
593 outContext->filledCircle = gdoFilledCircle;
594 outContext->arc = gdoArc;
595 outContext->dottedArc = gdoDottedArc;
596 outContext->setPen = gdoSetPen;
597 outContext->setBgPen = gdoSetBgPen;
598 outContext->setFontSize = gdoSetFontSize;
599 outContext->close = gdoClose;
600
601 return TRUE;
602}
