113static QByteArray stripComment(const QByteArray &s)
114{
115 // check if there is a comment at the end of the string
116 bool insideQuote=false;
117 int l = s.length();
118 for (int i=0;i<l;i++)
119 {
120 char c = s.at(i);
121 if (c=='\\') // skip over escaped characters
122 {
123 i++;
124 }
125 else if (c=='"') // toggle inside/outside quotation
126 {
127 insideQuote=!insideQuote;
128 }
129 else if (!insideQuote && c=='#') // found start of a comment
130 {
131 return s.left(i).trimmed();
132 }
133 }
134 return s;
135}

138static void processString()
139{
140 // strip leading and trailing whitespace
141 QByteArray s = stripComment(g_str.trimmed());
142 int l = s.length();
143
144 // remove surrounding quotes if present (and not escaped)
145 if (l>=2 && s.at(0)=='"' && s.at(l-1)=='"' && // remove quotes
146 (s.at(l-2)!='\\' || (s.at(l-2)=='\\' && s.at(l-3)=='\\')))
147 {
148 s=s.mid(1,s.length()-2);
149 l=s.length();
150 }
151
152 // check for invalid and/or escaped quotes
153 bool warned=false;
154 QByteArray result;
155 for (int i=0;i<l;i++)
156 {
157 char c = s.at(i);
158 if (c=='\\') // escaped character
159 {
160 if (i<l-1 && s.at(i+1)=='"') // unescape the quote character
161 {
162 result+='"';
163 }
164 else // keep other escaped characters in escaped form
165 {
166 result+=c;
167 if (i<l-1)
168 {
169 result+=s.at(i+1);
170 }
171 }
172 i++; // skip over the escaped character
173 }
174 else if (c=='"') // unescaped quote
175 {
176 if (!warned)
177 {
178 std::string str = g_str.trimmed().data();
179 config_warn("Invalid value for '%s' tag at line %d, file %s: Value '%s' is not properly quoted\n",
180 qPrintable(g_cmd),yylineno-1,qPrintable(g_yyFileName),str.c_str());
181 }
182 warned=true;
183 }
184 else // normal character
185 {
186 result+=c;
187 }
188 }
189
190 // recode the string
191 if (g_isEnum)
192 {
193 InputString *cur = dynamic_cast<InputString *>(g_curOption);
194 *g_arg = cur->checkEnumVal(g_codec->decode(result));
195 }
196 else
197 {
198 *g_arg = QVariant(g_codec->decode(result));
199 }
200
201 // update encoding
202 checkEncoding();
203
204 //printf("Processed string '%s'\n",g_string->data());
205}

207static void processList()
208{
209 bool allowCommaAsSeparator = g_cmd!=QString::fromLatin1("PREDEFINED");
210
211 const QByteArray s = stripComment(g_str.trimmed());
212 int l = s.length();
213
214 QByteArray elemStr;
215
216 // helper to push elemStr to the list and clear it
217 auto addElem = [&elemStr]()
218 {
219 if (!elemStr.isEmpty())
220 {
221 //printf("Processed list element '%s'\n",e.data());
222 *g_arg = QVariant(g_arg->toStringList() << g_codec->decode(elemStr));
223 elemStr="";
224 }
225 };
226
227 bool needsSeparator=false;
228 int insideQuote=false;
229 bool warned=false;
230 for (int i=0;i<l;i++)
231 {
232 char c = s.at(i);
233 if (!needsSeparator && c=='\\') // escaped character
234 {
235 if (i<l-1 && s.at(i+1)=='"') // unescape the quote character
236 {
237 elemStr+='"';
238 }
239 else // keep other escaped characters in escaped form
240 {
241 elemStr+=c;
242 if (i<l-1)
243 {
244 elemStr+=s.at(i+1);
245 }
246 }
247 i++; // skip over the escaped character
248 }
249 else if (!needsSeparator && c=='"') // quote character
250 {
251 if (!insideQuote)
252 {
253 insideQuote=true;
254 }
255 else // this quote ends an element
256 {
257 insideQuote=false;
258 needsSeparator=true;
259 }
260 }
261 else if (!insideQuote && ((c==',' && allowCommaAsSeparator) || isspace(c))) // separator
262 {
263 needsSeparator=false;
264 addElem();
265 }
266 else // normal content character
267 {
268 if (needsSeparator)
269 {
270 if (!warned)
271 {
272 std::string str = g_str.trimmed().data();
273 config_warn("Invalid value for '%s' tag at line %d, file %s: Values in list '%s' are not properly space %sseparated\n",
274 qPrintable(g_cmd),yylineno-1,qPrintable(g_yyFileName),str.c_str(),allowCommaAsSeparator?"or comma ":"");
275 warned=true;
276 }
277 needsSeparator=false;
278 i--; // try the character again as part of a new element
279 addElem();
280 }
281 else
282 {
283 elemStr+=c;
284 }
285 }
286 }
287 // add last part
288 addElem();
289 if (insideQuote)
290 {
291 std::string str = g_str.trimmed().data();
292 config_warn("Invalid value for '%s' tag at line %d, file %s: Values in list '%s' are not properly quoted\n",
293 qPrintable(g_cmd),yylineno-1,qPrintable(g_yyFileName),str.c_str());
294 }
295}

298static FILE *tryPath(const QString &path,const QString &fileName)
299{
300 QString absName=!path.isEmpty() ? path+QString::fromLatin1("/")+fileName : fileName;
301 QFileInfo fi(absName);
302 if (fi.exists() && fi.isFile())
303 {
304 FILE *f = fopen(absName.toLocal8Bit(),"r");
305 if (f==NULL)

313static FILE *findFile(const QString &fileName)
314{
315 if (QFileInfo(fileName).isAbsolute()) // absolute path
316 {
317 return tryPath(QString(), fileName);
318 }
319
320 // relative path, try with include paths in the list
321 QStringList sl = g_includePathList.toStringList();
322 substEnvVarsInStrList(sl);
323 foreach (QString s, sl)
324 {
325 FILE *f = tryPath(s,fileName);
326 if (f) return f;
327 }
328 // try cwd if g_includePathList fails
329 return tryPath(QString::fromLatin1("."),fileName);
330}

332static void readIncludeFile(const QString &incName)
333{
334 if (g_includeDepth==MAX_INCLUDE_DEPTH)
335 {
336 config_err("maximum include depth (%d) reached, %s is not included.",
337 MAX_INCLUDE_DEPTH,qPrintable(incName));
338 }
339
340 QString inc = incName;
341 substEnvVarsInString(inc);
342 inc = inc.trimmed();
343 uint incLen = inc.length();
344 if (inc.at(0)==QChar::fromLatin1('"') &&
345 inc.at(incLen-1)==QChar::fromLatin1('"')) // strip quotes
346 {
347 inc=inc.mid(1,incLen-2);
348 }
349
350 FILE *f = findFile(inc);
351 if (f) // see if the include file can be found
352 {
353 // For debugging
354#if SHOW_INCLUDES
355 for (i=0;i<includeStack.count();i++) msg(" ");
356 msg("@INCLUDE = %s: parsing...\n",qPrintable(inc));
357#endif
358
359 // store the state of the old file
360 ConfigFileState *fs=new ConfigFileState;
361 fs->oldState=YY_CURRENT_BUFFER;
362 fs->fileName=g_yyFileName;
363 fs->file=f;
364 // push the state on the stack
365 g_includeStack.push(fs);
366 // set the scanner to the include file
367 yy_switch_to_buffer(yy_create_buffer(f, YY_BUF_SIZE));
368 fs->newState=YY_CURRENT_BUFFER;
369 g_yyFileName=inc;
370 g_includeDepth++;
371 }
372 else
373 {
374 config_err("@INCLUDE = %s: not found!\n",qPrintable(inc));
375 }
376}

603static void substEnvVarsInString(QString &s)
604{
605 static QRegularExpression re(QString::fromLatin1("\\$\\([a-z_A-Z0-9]+\\)"));
606 if (s.isEmpty()) return;
607 int p=0;
608 int i,l;
609 //printf("substEnvVarInString(%s) start\n",qPrintable(s));
610
611 QRegularExpressionMatch match;
612 while ((i=s.indexOf(re,p,&match))!=-1)
613 {
614 l = match.capturedLength();
615 //printf("Found environment var s.mid(%d,%d)='%s'\n",i+2,l-3,qPrintable(s.mid(i+2,l-3)));
616 QString env=g_codec->decode(getenv(s.mid(i+2,l-3).toLatin1()));
617 substEnvVarsInString(env); // recursively expand variables if needed.
618 s = s.left(i)+env+s.right(s.length()-i-l);
619 p=i+env.length(); // next time start at the end of the expanded string
620 }
621 s=s.trimmed(); // to strip the bogus space that was added when an argument
622 // has quotes
623 //printf("substEnvVarInString(%s) end\n",qPrintable(s));
624}

626static void substEnvVarsInStrList(QStringList &sl)
627{
628 QStringList out;
629
630 foreach (QString result, sl)
631 {
632 // an argument with quotes will have an extra space at the end, so wasQuoted will be TRUE.
633 bool wasQuoted = (result.indexOf(QChar::fromLatin1(' '))!=-1) ||
634 (result.indexOf(QChar::fromLatin1('\t'))!=-1);
635 // here we strip the quote again
636 substEnvVarsInString(result);
637
638 //printf("Result %s was quoted=%d\n",qPrintable(result),wasQuoted);
639
640 if (!wasQuoted) /* as a result of the expansion, a single string
641 may have expanded into a list, which we'll
642 add to sl. If the original string already
643 contained multiple elements no further
644 splitting is done to allow quoted items with spaces! */
645 {
646 int l=result.length();
647 int i,p=0;
648 // skip spaces
649 // search for a "word"
650 for (i=0;i<l;i++)
651 {
652 QChar c;
653 // skip until start of new word
654 while (i<l && ((c=result.at(i))==QChar::fromLatin1(' ') || c==QChar::fromLatin1('\t'))) i++;
655 p=i; // p marks the start index of the word
656 // skip until end of a word
657 while (i<l && ((c=result.at(i))!=QChar::fromLatin1(' ') &&
658 c!=QChar::fromLatin1('\t') &&
659 c!=QChar::fromLatin1('"'))) i++;
660 if (i<l) // not at the end of the string
661 {
662 if (c==QChar::fromLatin1('"')) // word within quotes
663 {
664 p=i+1;
665 for (i++;i<l;i++)
666 {
667 c=result.at(i);
668 if (c==QChar::fromLatin1('"')) // end quote
669 {
670 out += result.mid(p,i-p);
671 p=i+1;
672 break;
673 }
674 else if (c==QChar::fromLatin1('\\')) // skip escaped stuff
675 {
676 i++;
677 }
678 }
679 }
680 else if (c==QChar::fromLatin1(' ') || c==QChar::fromLatin1('\t')) // separator
681 {
682 out += result.mid(p,i-p);
683 p=i+1;
684 }
685 }
686 }
687 if (p!=l) // add the leftover as a string
688 {
689 out += result.right(l-p);
690 }
691 }
692 else // just goto the next element in the list
693 {
694 out += result;
695 }
696 }
697 sl = out;
698}
